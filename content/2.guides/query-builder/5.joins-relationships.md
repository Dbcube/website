## JOIN Types

### join() - INNER JOIN

INNER JOIN returns only records that have matches in both tables:

```typescript
// Basic JOIN: Posts with author information
const postsWithAuthor = await db
  .table("posts")
  .join("users", "posts.user_id", "=", "users.id")
  .select([
    "posts.id",
    "posts.title",
    "posts.content",
    "users.name as author",
    "users.email as author_email",
  ])
  .get();

// JOIN with additional conditions
const publishedPostsWithAuthor = await db
  .table("posts")
  .join("users", "posts.user_id", "=", "users.id")
  .select(["posts.title", "users.name as author"])
  .where("posts.status", "published")
  .where("users.active", true)
  .get();
```

**Parameters:**

- `table` (string): Table to JOIN with
- `column1` (string): First column for JOIN condition
- `operator` (string): Comparison operator (usually `=`)
- `column2` (string): Second column for JOIN condition

### leftJoin() - LEFT OUTER JOIN

LEFT JOIN returns all records from the left table and matching ones from the right:

```typescript
// All users with their posts (even if they have no posts)
const usersWithPosts = await db
  .table("users")
  .leftJoin("posts", "users.id", "=", "posts.user_id")
  .select([
    "users.id",
    "users.name",
    "users.email",
    "posts.title as latest_post",
  ])
  .get();

// Count posts per user (including users without posts)
const postCountPerUser = await db
  .table("users")
  .leftJoin("posts", "users.id", "=", "posts.user_id")
  .select(["users.id", "users.name", db.raw("COUNT(posts.id) as total_posts")])
  .groupBy(["users.id", "users.name"])
  .get();

// Find users without posts
const usersWithoutPosts = await db
  .table("users")
  .leftJoin("posts", "users.id", "=", "posts.user_id")
  .select(["users.id", "users.name", "users.email"])
  .whereNull("posts.user_id")
  .get();
```

### rightJoin() - RIGHT OUTER JOIN

RIGHT JOIN returns all records from the right table and matching ones from the left:

```typescript
// All posts with user information (even if user was deleted)
const postsWithUsers = await db
  .table("users")
  .rightJoin("posts", "users.id", "=", "posts.user_id")
  .select([
    "posts.id",
    "posts.title",
    "posts.created_at",
    "users.name as author",
  ])
  .get();
```

### fullOuterJoin() - FULL OUTER JOIN

FULL OUTER JOIN returns records when there are matches in either table:

```typescript
// All users and posts (some users without posts, some posts without user)
const allData = await db
  .table("users")
  .fullOuterJoin("posts", "users.id", "=", "posts.user_id")
  .select(["users.name", "posts.title"])
  .get();
```

### crossJoin() - CROSS JOIN

CROSS JOIN produces the Cartesian product of the tables:

```typescript
// All combinations of colors and sizes
const combinations = await db
  .table("colors")
  .crossJoin("sizes")
  .select(["colors.name as color", "sizes.name as size"])
  .get();

// Useful for generating product catalogs
const productVariations = await db
  .table("products")
  .crossJoin("colors")
  .crossJoin("sizes")
  .select([
    "products.name as product",
    "colors.name as color",
    "sizes.name as size",
    db.raw(
      'CONCAT(products.name, " - ", colors.name, " - ", sizes.name) as sku'
    ),
  ])
  .where("products.active", true)
  .get();
```

## Multiple JOINs

### Chaining Multiple JOINs

```typescript
// Posts with author, category and tags
const completeData = await db
  .table("posts")
  .join("users", "posts.user_id", "=", "users.id")
  .join("categories", "posts.category_id", "=", "categories.id")
  .leftJoin("post_tags", "posts.id", "=", "post_tags.post_id")
  .leftJoin("tags", "post_tags.tag_id", "=", "tags.id")
  .select([
    "posts.id",
    "posts.title",
    "posts.content",
    "users.name as author",
    "users.email as author_email",
    "categories.name as category",
    "tags.name as tag",
  ])
  .where("posts.status", "published")
  .get();

// E-commerce: Products with category, brand and reviews
const detailedProducts = await db
  .table("products")
  .join("categories", "products.category_id", "=", "categories.id")
  .join("brands", "products.brand_id", "=", "brands.id")
  .leftJoin("reviews", "products.id", "=", "reviews.product_id")
  .select([
    "products.id",
    "products.name",
    "products.price",
    "categories.name as category",
    "brands.name as brand",
    db.raw("AVG(reviews.rating) as average_rating"),
    db.raw("COUNT(reviews.id) as total_reviews"),
  ])
  .where("products.active", true)
  .groupBy([
    "products.id",
    "products.name",
    "products.price",
    "categories.name",
    "brands.name",
  ])
  .get();
```

### JOINs with Complex Conditions

```typescript
// JOIN with multiple conditions
const specialPosts = await db
  .table("posts")
  .join("users", function () {
    this.on("posts.user_id", "=", "users.id")
      .andOn("users.active", "=", db.raw("true"))
      .andOn("users.verified", "=", db.raw("true"));
  })
  .select(["posts.title", "users.name as verified_author"])
  .get();

// JOIN with additional WHERE conditions
const collaborations = await db
  .table("posts")
  .join("users as authors", "posts.author_id", "=", "authors.id")
  .join("users as editors", function () {
    this.on("posts.editor_id", "=", "editors.id")
      .andOn("editors.role", "=", db.raw('"editor"'))
      .andOn("editors.id", "!=", "posts.author_id");
  })
  .select(["posts.title", "authors.name as author", "editors.name as editor"])
  .get();
```

## JOINs with Subqueries

### JOIN with Subquery as Table

```typescript
// JOIN with calculated statistics
const usersWithStats = await db
  .table("users")
  .join(
    db
      .table("posts")
      .select([
        "user_id",
        db.raw("COUNT(*) as total_posts"),
        db.raw("AVG(views) as average_views"),
      ])
      .groupBy("user_id")
      .as("stats"),
    "users.id",
    "=",
    "stats.user_id"
  )
  .select([
    "users.name",
    "users.email",
    "stats.total_posts",
    "stats.average_views",
  ])
  .get();

// Products with latest review
const productsWithLatestReview = await db
  .table("products")
  .join(
    db
      .table("reviews")
      .select(["product_id", "comment", "rating", "created_at"])
      .whereIn("id", function () {
        this.select(db.raw("MAX(id)"))
          .from("reviews as r2")
          .whereRaw("r2.product_id = reviews.product_id")
          .groupBy("product_id");
      })
      .as("latest_review"),
    "products.id",
    "=",
    "latest_review.product_id"
  )
  .select([
    "products.name",
    "latest_review.comment",
    "latest_review.rating",
    "latest_review.created_at as latest_review_date",
  ])
  .get();
```

## Complex Relationships

### Many-to-Many Relationships

```typescript
// Posts with their tags (N:M relationship)
const postsWithTags = await db
  .table("posts")
  .join("post_tags", "posts.id", "=", "post_tags.post_id")
  .join("tags", "post_tags.tag_id", "=", "tags.id")
  .select(["posts.id as post_id", "posts.title", "tags.name as tag"])
  .where("posts.status", "published")
  .get();

// Group tags by post
async function getPostsWithGroupedTags() {
  const data = await db
    .table("posts")
    .join("post_tags", "posts.id", "=", "post_tags.post_id")
    .join("tags", "post_tags.tag_id", "=", "tags.id")
    .select([
      "posts.id",
      "posts.title",
      "posts.content",
      db.raw("GROUP_CONCAT(tags.name) as tags"),
    ])
    .where("posts.status", "published")
    .groupBy(["posts.id", "posts.title", "posts.content"])
    .get();

  // Process tags as array
  return data.map((post) => ({
    ...post,
    tags: post.tags ? post.tags.split(",") : [],
  }));
}
```

### Self-JOIN

```typescript
// Hierarchical structure: Categories with subcategories
const categoriesWithParents = await db
  .table("categories as c1")
  .leftJoin("categories as c2", "c1.parent_id", "=", "c2.id")
  .select(["c1.id", "c1.name", "c2.name as parent_category"])
  .get();

// Employees with their managers
const employeesWithManagers = await db
  .table("employees as e1")
  .leftJoin("employees as e2", "e1.manager_id", "=", "e2.id")
  .select(["e1.name as employee", "e1.position", "e2.name as manager"])
  .get();

// Find all subordinates of a manager
async function getSubordinates(managerId: number) {
  return await db
    .table("employees as e1")
    .join("employees as e2", "e1.manager_id", "=", "e2.id")
    .select(["e1.id", "e1.name", "e1.position", "e2.name as manager"])
    .where("e2.id", managerId)
    .get();
}
```

## JOIN Optimization

### Use Appropriate Indexes

```typescript
// ✅ Good - JOIN columns are indexed
const result = await db
  .table("posts")
  .join("users", "posts.user_id", "=", "users.id") // Ensure user_id and id are indexed
  .select(["posts.title", "users.name"])
  .get();

// ✅ Good - filter before JOIN when possible
const recentPosts = await db
  .table("posts")
  .join("users", "posts.user_id", "=", "users.id")
  .where("posts.created_at", ">", "2023-01-01") // Filter on main table first
  .where("users.active", true)
  .get();
```

### Avoid Unnecessary JOINs

```typescript
// ❌ Avoid - unnecessary JOIN if only counting
const postCountPerUser = await db
  .table("posts")
  .join("users", "posts.user_id", "=", "users.id")
  .select(["users.name", db.raw("COUNT(*) as total_posts")])
  .groupBy("users.id")
  .get();

// ✅ Better - use subquery if more efficient
const usersWithCount = await db
  .table("users")
  .select([
    "name",
    db
      .table("posts")
      .count("*")
      .where("user_id", db.raw("users.id"))
      .as("total_posts"),
  ])
  .get();
```

## Practical Examples

### 1. Complete Blog System

```typescript
async function getCompletePostsForBlog(limit: number = 10) {
  return await db
    .table("posts")
    .join("users", "posts.user_id", "=", "users.id")
    .join("categories", "posts.category_id", "=", "categories.id")
    .leftJoin("images", "posts.image_id", "=", "images.id")
    .select([
      "posts.id",
      "posts.title",
      "posts.excerpt",
      "posts.content",
      "posts.published_at",
      "posts.views",
      "users.name as author",
      "users.avatar as author_avatar",
      "categories.name as category",
      "categories.slug as category_slug",
      "categories.color as category_color",
      "images.url as featured_image",
      "images.alt as image_alt",

      // Subqueries for additional data
      db
        .table("comments")
        .count("*")
        .where("post_id", db.raw("posts.id"))
        .where("approved", true)
        .as("total_comments"),

      db
        .table("likes")
        .count("*")
        .where("post_id", db.raw("posts.id"))
        .as("total_likes"),
    ])
    .where("posts.status", "published")
    .where("posts.published_at", "<=", db.raw("NOW()"))
    .orderBy("posts.published_at", "desc")
    .limit(limit)
    .get();
}
```

### 2. E-commerce Dashboard

```typescript
async function getSalesDashboard() {
  const [salesByCategory, topProducts, vipCustomers, recentOrders] =
    await Promise.all([
      // Sales by category
      db
        .table("order_items")
        .join("products", "order_items.product_id", "=", "products.id")
        .join("categories", "products.category_id", "=", "categories.id")
        .join("orders", "order_items.order_id", "=", "orders.id")
        .select([
          "categories.name as category",
          db.raw("SUM(order_items.quantity) as products_sold"),
          db.raw("SUM(order_items.subtotal) as total_revenue"),
        ])
        .where("orders.status", "completed")
        .whereRaw("orders.created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)")
        .groupBy(["categories.id", "categories.name"])
        .orderBy("total_revenue", "desc")
        .get(),

      // Top products
      db
        .table("order_items")
        .join("products", "order_items.product_id", "=", "products.id")
        .join("orders", "order_items.order_id", "=", "orders.id")
        .select([
          "products.name",
          "products.price",
          db.raw("SUM(order_items.quantity) as quantity_sold"),
          db.raw("SUM(order_items.subtotal) as revenue"),
        ])
        .where("orders.status", "completed")
        .whereRaw("orders.created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)")
        .groupBy(["products.id", "products.name", "products.price"])
        .orderBy("quantity_sold", "desc")
        .limit(10)
        .get(),

      // VIP customers
      db
        .table("orders")
        .join("users", "orders.user_id", "=", "users.id")
        .select([
          "users.name",
          "users.email",
          db.raw("COUNT(orders.id) as total_orders"),
          db.raw("SUM(orders.total) as total_spent"),
          db.raw("AVG(orders.total) as average_order"),
        ])
        .where("orders.status", "completed")
        .groupBy(["users.id", "users.name", "users.email"])
        .having("total_spent", ">", 1000)
        .orderBy("total_spent", "desc")
        .limit(20)
        .get(),

      // Recent orders with details
      db
        .table("orders")
        .join("users", "orders.user_id", "=", "users.id")
        .leftJoin(
          "addresses",
          "orders.shipping_address_id",
          "=",
          "addresses.id"
        )
        .select([
          "orders.id",
          "orders.order_number",
          "orders.total",
          "orders.status",
          "orders.created_at",
          "users.name as customer",
          "users.email as customer_email",
          "addresses.city",
          "addresses.country",
        ])
        .orderBy("orders.created_at", "desc")
        .limit(50)
        .get(),
    ]);

  return {
    salesByCategory: salesByCategory,
    topProducts: topProducts,
    vipCustomers: vipCustomers,
    recentOrders: recentOrders,
  };
}
```

### 3. Permission System with Roles

```typescript
async function checkUserPermission(
  userId: number,
  resource: string,
  action: string
): Promise<boolean> {
  const permissions = await db
    .table("users")
    .join("user_roles", "users.id", "=", "user_roles.user_id")
    .join("roles", "user_roles.role_id", "=", "roles.id")
    .join("role_permissions", "roles.id", "=", "role_permissions.role_id")
    .join(
      "permissions",
      "role_permissions.permission_id",
      "=",
      "permissions.id"
    )
    .select("permissions.resource", "permissions.action")
    .where("users.id", userId)
    .where("users.active", true)
    .where("permissions.resource", resource)
    .where("permissions.action", action)
    .get();

  return permissions.length > 0;
}

// Get all user permissions
async function getUserPermissions(userId: number) {
  return await db
    .table("users")
    .join("user_roles", "users.id", "=", "user_roles.user_id")
    .join("roles", "user_roles.role_id", "=", "roles.id")
    .join("role_permissions", "roles.id", "=", "role_permissions.role_id")
    .join(
      "permissions",
      "role_permissions.permission_id",
      "=",
      "permissions.id"
    )
    .select([
      "roles.name as role",
      "permissions.resource",
      "permissions.action",
      "permissions.description",
    ])
    .where("users.id", userId)
    .where("users.active", true)
    .distinct()
    .get();
}
```
