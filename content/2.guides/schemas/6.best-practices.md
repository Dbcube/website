## File Organization

### Recommended Structure

```
project/
└── .dbcube/
    ├── schemas/                    # Table definitions
    │   ├── core/                   # Fundamental tables
    │   │   ├── users.table.cube
    │   │   ├── roles.table.cube
    │   │   ├── permissions.table.cube
    │   │   └── settings.table.cube
    │   ├── auth/                   # Authentication and authorization
    │   │   ├── sessions.table.cube
    │   │   ├── tokens.table.cube
    │   │   └── oauth_providers.table.cube
    │   ├── content/                # Content management
    │   │   ├── posts.table.cube
    │   │   ├── categories.table.cube
    │   │   ├── tags.table.cube
    │   │   ├── comments.table.cube
    │   │   └── media.table.cube
    │   ├── ecommerce/              # E-commerce (if applicable)
    │   │   ├── products.table.cube
    │   │   ├── product_categories.table.cube
    │   │   ├── orders.table.cube
    │   │   ├── order_items.table.cube
    │   │   └── payments.table.cube
    │   └── analytics/              # Analytics and metrics
    │       ├── events.table.cube
    │       ├── metrics.table.cube
    │       └── reports.table.cube
    ├── seeders/                    # Initial data ordered
    │   ├── 01-settings.seeder.cube        # Base configuration
    │   ├── 02-roles.seeder.cube           # System roles
    │   ├── 03-admin-users.seeder.cube     # Administrative users
    │   ├── 04-categories.seeder.cube      # Main categories
    │   ├── 05-sample-content.seeder.cube  # Demo content
    │   ├── 90-development.seeder.cube     # Development-only data
    │   └── 99-testing.seeder.cube         # Test data
    └── triggers/                   # Business logic
        ├── audit/                  # Audit triggers
        │   ├── general-audit.trigger.cube
        │   └── critical-audit.trigger.cube
        ├── validations/            # Validation triggers
        │   ├── users.trigger.cube
        │   ├── content.trigger.cube
        │   └── commerce.trigger.cube
        ├── notifications/          # Notification system
        │   ├── email.trigger.cube
        │   └── push.trigger.cube
        └── business-logic/         # Specific business logic
            ├── inventory.trigger.cube
            ├── billing.trigger.cube
            └── analytics.trigger.cube
```

### Naming Conventions

#### Files and Directories

::cube-code
---
code: |
    // ✅ Descriptive and consistent names
    users.table.cube           // Singular for tables
    posts.table.cube           // Plural if context requires it
    user-profiles.table.cube   // Hyphens for compound names

    // ✅ Numbered seeders by dependency order
    01-roles.seeder.cube       // No dependencies
    02-users.seeder.cube       // Depends on roles
    03-posts.seeder.cube       // Depends on users

    // ✅ Triggers grouped by functionality
    validations.trigger.cube    // Multiple validation triggers
    audit-users.trigger.cube   // User-specific audit triggers
---
::

#### Table and Column Names

::cube-code
---
code: |
    // ✅ Use snake_case consistently
    @meta({
        name: "user_profiles";        // Not "userProfiles" or "user-profiles"
    });

    @columns({
        user_id: { type: "int"; };       // Not "userId" or "user-id"
        created_at: { type: "timestamp"; };  // Not "createdAt"
        is_active: { type: "boolean"; }; // Not "isActive" or "active"
    });
---
::

#### Foreign Keys and Relationships

::cube-code
---
code: |
    @columns({
        // ✅ Consistent pattern: singular_table + _id
        user_id: {
            type: "int";
            foreign: { table: "users"; column: "id"; };
        };

        category_id: {
            type: "int";
            foreign: { table: "categories"; column: "id"; };
        };

        // ✅ For self-references, be specific
        parent_category_id: {
            type: "int";
            foreign: { table: "categories"; column: "id"; };
        };

        parent_comment_id: {
            type: "int";
            foreign: { table: "comments"; column: "id"; };
        };
    });
---
::

## Recommended Design Patterns

### 1. Standard Fields in All Tables

::cube-code
---
code: |
    @columns({
        // Standard primary key
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
            description: "Unique identifier";
        };

        // Standard timestamps (recommended for all tables)
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
            description: "Record creation timestamp";
        };

        updated_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP";
            options: ["not null"];
            description: "Last update timestamp";
        };

        // For soft deletes (optional but recommended)
        deleted_at: {
            type: "timestamp";
            description: "Logical deletion timestamp (NULL = active)";
        };

        // For advanced auditing (optional)
        created_by: {
            type: "int";
            options: ["index"];
            foreign: { table: "users"; column: "id"; };
            description: "User who created the record";
        };

        updated_by: {
            type: "int";
            options: ["index"];
            foreign: { table: "users"; column: "id"; };
            description: "User who updated the record";
        };
    });
---
::

### 2. Status Management with Enums

::cube-code
---
code: |
    @columns({
        // ✅ Clear and extensible statuses
        status: {
            type: "enum";
            enumValues: [
                "draft",        // Initial
                "review",       // In process
                "published",    // Active
                "archived",     // Temporarily inactive
                "deleted"       // Marked for deletion
            ];
            defaultValue: "draft";
            options: ["not null"];
            description: "Content status";
        };

        // For complex states, use JSON
        approval_workflow: {
            type: "json";
            defaultValue: {
                "current_status": "pending",
                "required_approvals": 2,
                "received_approvals": 0,
                "history": []
            };
            description: "Approval workflow state";
        };
    });
---
::

### 3. Flexible Metadata with JSON

::cube-code
---
code: |
    @columns({
        // For flexible configurations
        settings: {
            type: "json";
            defaultValue: {};
            description: "Record-specific settings";
        };

        // For analytics data
        metrics: {
            type: "json";
            defaultValue: {
                "views": 0,
                "likes": 0,
                "shares": 0,
                "average_read_time": 0

        // For file metadata
        file_properties: {
            type: "json";
            defaultValue: {};
            description: "File properties (size, type, dimensions, etc.)";
        };
    });
---
::

## Patterns for Complex Relationships

### 1. Hierarchies (Self-referencing)

::cube-code
---
code: |
    // categories.table.cube
    @database("content");

    @meta({
        name: "categories";
        description: "Hierarchical categories for content";
    });

    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };
        name: {
            type: "varchar";
            length: "100";
            options: ["not null"];
        };
        slug: {
            type: "varchar";
            length: "100";
            options: ["not null", "unique"];
        };

        // Self-reference for hierarchy
        parent_category_id: {
            type: "int";
            options: ["index"];
            foreign: { table: "categories"; column: "id"; };
            description: "Parent category (NULL for root categories)";
        };

        // Useful fields for hierarchies
        level: {
            type: "int";
            defaultValue: 0;
            options: ["unsigned"];
            description: "Hierarchy depth level (0 = root)";
        };

        path: {
            type: "varchar";
            length: "500";
            description: "Full path: /parent/child/grandchild";
        };

        sort_order: {
            type: "int";
            defaultValue: 0;
            description: "Display order among siblings";
        };

        active: {
            type: "boolean";
            defaultValue: true;
            options: ["not null"];
        };
    });
---
::

### 2. Polymorphic Relationships

::cube-code
---
code: |
    // For when one model can belong to multiple parent model types
    // Example: Comments that can be on posts, products, etc.

    // comments.table.cube
    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };
        content: {
            type: "text";
            options: ["not null"];
        };

        // Polymorphic relationship
        commentable_id: {
            type: "int";
            options: ["not null", "index"];
            description: "ID of parent record (post, product, etc.)";
        };

        commentable_type: {
            type: "varchar";
            length: "50";
            options: ["not null", "index"];
            description: "Type of parent model (posts, products, etc.)";
        };

        user_id: {
            type: "int";
            options: ["not null", "index"];
            foreign: { table: "users"; column: "id"; };
        };
    });
---
::

### 3. Pivot Tables with Additional Data

::cube-code
---
code: |
    // user_roles.table.cube - Enriched pivot table
    @database("auth");

    @meta({
        name: "user_roles";
        description: "User-role relationship with additional metadata";
    });

    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };

        user_id: {
            type: "int";
            options: ["not null", "index"];
            foreign: { table: "users"; column: "id"; };
        };

        role_id: {
            type: "int";
            options: ["not null", "index"];
            foreign: { table: "roles"; column: "id"; };
        };

        // Additional relationship data
        active: {
            type: "boolean";
            defaultValue: true;
            options: ["not null"];
            description: "Whether the role is active for the user";
        };

        start_date: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
            description: "When the role was assigned";
        };

        end_date: {
            type: "timestamp";
            description: "When the role expires (NULL = permanent)";
        };

        assigned_by: {
            type: "int";
            options: ["index"];
            foreign: { table: "users"; column: "id"; };
            description: "Who assigned the role";
        };

        context: {
            type: "json";
            defaultValue: {};
            description: "Role-specific context (project, department, etc.)";
        };

        // Composite index would be validated in triggers
        // to ensure no duplicate active user_id + role_id combinations
    });
---
::
