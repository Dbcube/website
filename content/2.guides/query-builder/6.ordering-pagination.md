## Result Ordering

### orderBy() - Basic Ordering

```typescript
// Ascending order (default)
const users = await db.table("users").orderBy("name").get();

// Descending order
const recentPosts = await db.table("posts").orderBy("created_at", "desc").get();

// Multiple ordering criteria
const orderedUsers = await db
  .table("users")
  .orderBy("active", "desc") // Active users first
  .orderBy("created_at", "desc") // Most recent first
  .orderBy("name", "asc") // Alphabetically
  .get();
```

### orderByRaw() - Custom Ordering

```typescript
// Order by SQL expression
const products = await db
  .table("products")
  .orderByRaw("price * discount DESC")
  .get();

// Order by CASE WHEN
const tasks = await db
  .table("tasks")
  .orderByRaw(
    `
    CASE priority 
      WHEN 'high' THEN 1
      WHEN 'medium' THEN 2
      WHEN 'low' THEN 3
    END ASC
  `
  )
  .get();
```

## Result Limiting

### limit() - Limit Number of Records

```typescript
// Get the 10 most recent posts
const latestPosts = await db
  .table("posts")
  .orderBy("created_at", "desc")
  .limit(10)
  .get();

// Top 5 best-selling products
const topProducts = await db
  .table("products")
  .orderBy("total_sales", "desc")
  .limit(5)
  .get();
```

### offset() - Skip Records

```typescript
// Page 2: skip the first 10 and take the next 10
const page2 = await db
  .table("users")
  .orderBy("name")
  .limit(10)
  .offset(10)
  .get();

// Page 3: skip the first 20
const page3 = await db
  .table("users")
  .orderBy("name")
  .limit(10)
  .offset(20)
  .get();
```

## Advanced Pagination

### Pagination with Total Count

```typescript
interface PaginationResult<T> {
  data: T[];
  pagination: {
    page: number;
    perPage: number;
    total: number;
    totalPages: number;
    hasNext: boolean;
    hasPrev: boolean;
  };
}

async function paginateQuery<T>(
  query: any,
  page: number = 1,
  perPage: number = 20
): Promise<PaginationResult<T>> {
  const offset = (page - 1) * perPage;

  // Clone query for counting
  const countQuery = query.clone();

  // Execute both queries in parallel
  const [data, total] = await Promise.all([
    query.limit(perPage).offset(offset).get(),
    countQuery.count(),
  ]);

  const totalPages = Math.ceil(total / perPage);

  return {
    data,
    pagination: {
      page,
      perPage,
      total,
      totalPages,
      hasNext: page < totalPages,
      hasPrev: page > 1,
    },
  };
}

// Usage of pagination
const result = await paginateQuery(
  db
    .table("products")
    .select(["id", "name", "price"])
    .where("active", true)
    .orderBy("name"),
  2, // page
  15 // per page
);
```

### Cursor-based Pagination (More Efficient)

```typescript
// Pagination using ID as cursor
async function paginateByCursor(
  table: string,
  lastId: number = 0,
  limit: number = 20
) {
  const records = await db
    .table(table)
    .where("id", ">", lastId)
    .orderBy("id")
    .limit(limit + 1) // +1 to check if there are more
    .get();

  const hasMore = records.length > limit;
  const data = hasMore ? records.slice(0, -1) : records;
  const nextCursor = data.length > 0 ? data[data.length - 1].id : null;

  return {
    data,
    nextCursor,
    hasMore,
  };
}

// Usage
let cursor = 0;
while (true) {
  const result = await paginateByCursor("posts", cursor, 50);

  // Process data
  console.log(`Processing ${result.data.length} posts`);

  if (!result.hasMore) break;
  cursor = result.nextCursor;
}
```

## Practical Examples

### 1. Product List with Filters and Ordering

```typescript
interface ProductFilters {
  category?: number;
  priceMin?: number;
  priceMax?: number;
  sortBy?: "name" | "price" | "date" | "popularity";
  sortDirection?: "asc" | "desc";
  page?: number;
  perPage?: number;
}

async function getFilteredProducts(filters: ProductFilters = {}) {
  let query = db
    .table("products")
    .select([
      "products.id",
      "products.name",
      "products.price",
      "products.image",
      "products.created_at",
      "categories.name as category",
      db.raw("AVG(reviews.rating) as average_rating"),
      db.raw("COUNT(reviews.id) as total_reviews"),
    ])
    .leftJoin("categories", "products.category_id", "=", "categories.id")
    .leftJoin("reviews", "products.id", "=", "reviews.product_id")
    .where("products.active", true)
    .groupBy([
      "products.id",
      "products.name",
      "products.price",
      "products.image",
      "products.created_at",
      "categories.name",
    ]);

  // Apply filters
  if (filters.category) {
    query = query.where("products.category_id", filters.category);
  }

  if (filters.priceMin) {
    query = query.where("products.price", ">=", filters.priceMin);
  }

  if (filters.priceMax) {
    query = query.where("products.price", "<=", filters.priceMax);
  }

  // Apply ordering
  const sortBy = filters.sortBy || "name";
  const direction = filters.sortDirection || "asc";

  switch (sortBy) {
    case "name":
      query = query.orderBy("products.name", direction);
      break;
    case "price":
      query = query.orderBy("products.price", direction);
      break;
    case "date":
      query = query.orderBy("products.created_at", direction);
      break;
    case "popularity":
      query = query.orderBy("total_reviews", direction);
      break;
  }

  // Apply pagination
  return await paginateQuery(query, filters.page || 1, filters.perPage || 20);
}
```

### 2. Activity Timeline with Pagination

```typescript
async function getActivityTimeline(
  userId?: number,
  lastDate?: string,
  limit: number = 20
) {
  let query = db
    .table("activities")
    .select([
      "activities.id",
      "activities.type",
      "activities.description",
      "activities.created_at",
      "activities.extra_data",
      "users.name as user",
      "users.avatar",
    ])
    .join("users", "activities.user_id", "=", "users.id");

  // Filter by user if specified
  if (userId) {
    query = query.where("activities.user_id", userId);
  }

  // Date-based pagination (cursor)
  if (lastDate) {
    query = query.where("activities.created_at", "<", lastDate);
  }

  const activities = await query
    .orderBy("activities.created_at", "desc")
    .limit(limit + 1)
    .get();

  const hasMore = activities.length > limit;
  const data = hasMore ? activities.slice(0, -1) : activities;

  return {
    activities: data,
    nextDate: data.length > 0 ? data[data.length - 1].created_at : null,
    hasMore,
  };
}
```

### 3. Ranking with Pagination

```typescript
async function getUserRanking(page: number = 1, perPage: number = 50) {
  const query = db
    .table("users")
    .select([
      "users.id",
      "users.name",
      "users.avatar",
      "users.total_points",
      db.raw("RANK() OVER (ORDER BY total_points DESC) as position"),
      db.raw("COUNT(user_achievements.achievement_id) as total_achievements"),
    ])
    .leftJoin("user_achievements", "users.id", "=", "user_achievements.user_id")
    .where("users.active", true)
    .groupBy(["users.id", "users.name", "users.avatar", "users.total_points"])
    .orderBy("users.total_points", "desc");

  return await paginateQuery(query, page, perPage);
}
```

## Ordering and Pagination Optimization

### 1. Indexes for Ordering

```sql
-- Create indexes to improve ORDER BY
CREATE INDEX idx_users_name ON users(name);
CREATE INDEX idx_posts_date_status ON posts(created_at, status);
CREATE INDEX idx_products_category_price ON products(category_id, price);
```

### 2. Avoid OFFSET in Large Datasets

```typescript
// ❌ Problematic for large offsets
const page1000 = await db
  .table("users")
  .orderBy("id")
  .limit(20)
  .offset(20000) // Very slow
  .get();

// ✅ Better - use cursor
const nextPage = await db
  .table("users")
  .where("id", ">", lastId)
  .orderBy("id")
  .limit(20)
  .get();
```

### 3. Optimize Queries with Count

```typescript
// ✅ Optimize count with appropriate indexes
async function optimizedPaginate(filters: any, page: number, perPage: number) {
  // Build base query
  let baseQuery = db.table("products").where("active", true);

  if (filters.category) {
    baseQuery = baseQuery.where("category_id", filters.category);
  }

  // Data query with complete joins
  const dataQuery = baseQuery
    .clone()
    .select(["products.*", "categories.name as category"])
    .join("categories", "products.category_id", "=", "categories.id");

  // Simpler count query (without unnecessary joins)
  const countQuery = baseQuery.clone();

  const [data, total] = await Promise.all([
    dataQuery
      .orderBy("products.name")
      .limit(perPage)
      .offset((page - 1) * perPage)
      .get(),
    countQuery.count(),
  ]);

  return { data, total };
}
```
