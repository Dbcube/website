## What are Relationships?

Relationships connect tables through foreign keys, enabling:

- **Referential Integrity**: Ensure related data exists
- **Efficient Queries**: Perform optimized JOINs between tables
- **Scalability**: Normalize data to avoid duplication
- **Consistency**: Maintain data coherence

## Types of Relationships

### 1. One-to-Many Relationship (1:N)

The most common relationship: one record in the parent table can have many records in the child table.

::cube-code
---
code: |
    // categories.table.cube
    @database("blog");

    @meta({
        name: "categories";
        description: "Categories to organize posts";
    });

    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };
        name: {
            type: "varchar";
            length: "100";
            options: ["not null", "unique"];
        };
        slug: {
            type: "varchar";
            length: "100";
            options: ["not null", "unique"];
        };
        description: {
            type: "text";
        };
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
        };
    });
---
::

::cube-code
---
code: |
    // posts.table.cube
    @database("blog");

    @meta({
        name: "posts";
        description: "Blog posts with category relationship";
    });

    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };
        title: {
            type: "varchar";
            length: "255";
            options: ["not null"];
        };
        content: {
            type: "text";
            options: ["not null"];
        };
        // 🔗 Foreign key to categories
        category_id: {
            type: "int";
            options: ["not null", "index"];
            foreign: {
                table: "categories";
                column: "id";
            };
            description: "Category this post belongs to";
        };
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
        };
    });
---

### 2. One-to-One Relationship (1:1)

One record in a table relates to exactly one record in another table.

            type: "varchar";
            length: "255";
            options: ["not null", "unique"];
        };
        password: {
            type: "varchar";
            length: "255";
            options: ["not null"];
        };
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];

### 3. Many-to-Many Relationship (N:M)

Implemented using an intermediate (pivot) table that connects two tables.

::cube-code
---
code: |
    // users.table.cube (already defined above)

    // roles.table.cube
    @database("app");

    @meta({
        name: "roles";
        description: "System roles for access control";
    });

    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };
        name: {
            type: "varchar";
            length: "50";
            options: ["not null", "unique"];
        };
        description: {
            type: "varchar";
            length: "255";
        };
        permissions: {
            type: "json";
            defaultValue: [];
            description: "Array of role permissions";
        };
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
        };
    });
---
::

::cube-code
---
code: |
    // user_roles.table.cube - Pivot table
    @database("app");

    @meta({
        name: "user_roles";
        description: "Pivot table for many-to-many relationship between users and roles";
    });

    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };
        // 🔗 Foreign key to users
        user_id: {
            type: "int";
            options: ["not null", "index"];
            foreign: {
                table: "users";
                column: "id";
            };
            description: "User ID";
        };
        // 🔗 Foreign key to roles
        role_id: {
            type: "int";
            options: ["not null", "index"];
            foreign: {
                table: "roles";
                column: "id";
            };
            description: "Role ID";
        };
        active: {
            type: "boolean";
            defaultValue: true;
            options: ["not null"];
            description: "Whether the role is active for the user";
        };
        assigned_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
        };
    });
---

## Foreign Key Definition

### Basic Syntax

::cube-code
---
code: |
    @columns({
        foreign_key_column: {
            type: "int";
            options: ["not null", "index"];
            foreign: {
                table: "referenced_table";
                column: "referenced_column";
            };
            description: "Description of the relationship";
        };
    });
---
::

### Foreign Key Properties

::cube-code
---
code: |
    @columns({
        category_id: {
            type: "int";                    // Type must match referenced column
            options: [
                "not null",                 // Usually required
                "index"                     // Always create index for performance
            ];
            foreign: {
                table: "categories";        // Referenced table name
                column: "id";               // Referenced column (usually 'id')
            };
            description: "Relationship to post category";
        };
    });
---
::

## Best Practices

### 1. Foreign Key Naming

::cube-code
---
code: |
    @columns({
        // ✅ Use pattern: singular_table + _id
        user_id: {
            type: "int";
            foreign: { table: "users"; column: "id"; };
        };

        category_id: {
            type: "int";
            foreign: { table: "categories"; column: "id"; };
        };

        // ❌ Avoid ambiguous names
        user: { type: "int"; };           // Not clear it's a foreign key
        category: { type: "int"; };       // Not clear it's a foreign key
        id_user: { type: "int"; };        // Less common convention
    });
---
::

### 2. Always Index Foreign Keys

::cube-code
---
code: |
    @columns({
        // ✅ Include index on foreign keys for better performance
        category_id: {
            type: "int";
            options: ["not null", "index"];  // 'index' is crucial for fast JOINs
            foreign: {
                table: "categories";
                column: "id";
            };
        };
    });
---
::

### 3. Use NOT NULL When Appropriate

::cube-code
---
code: |
    @columns({
        // ✅ NOT NULL for required relationships
        user_id: {
            type: "int";
            options: ["not null", "index"];
            foreign: { table: "users"; column: "id"; };
        };

        // ✅ Allow NULL for optional relationships
        category_id: {
            type: "int";
            options: ["index"];  // Without "not null"
            foreign: { table: "categories"; column: "id"; };
            description: "Optional category";
        };
    });
---
::

### 4. Document Relationships

::cube-code

code: |

## Common Validation Errors

### 1. Type Mismatch

::cube-code
---
code: |
    // ❌ Types don't match
    // users.id is INT
    user_id: {
        type: "varchar";  // Error: should be int
        foreign: { table: "users"; column: "id"; };
    };

    // ✅ Types match
    user_id: {
        type: "int";     // Correct: same type as users.id
        foreign: { table: "users"; column: "id"; };
    };
---
::

### 2. Non-existent Table or Column

::cube-code
---
code: |
    // ❌ Reference to non-existent table
    category_id: {
        type: "int";
        foreign: { table: "categories"; column: "id"; };  // Error: table doesn't exist
    };

    // ✅ Correct reference
    category_id: {
        type: "int";
        foreign: { table: "categories"; column: "id"; };  // Correct
    };
---
::

### 3. Missing Index

::cube-code
---
code: |
    // ❌ Foreign key without index (bad for performance)
    user_id: {
        type: "int";
        options: ["not null"];  // Missing "index"
        foreign: { table: "users"; column: "id"; };
    };

    // ✅ Foreign key with index
    user_id: {
        type: "int";
        options: ["not null", "index"];  // Includes index
        foreign: { table: "users"; column: "id"; };
    };
---
::

## Querying with Relationships

### Using Dbcube ORM

```typescript
// Query with automatic JOIN
const posts = await db
  .posts()
  .with(["user", "category"]) // Load relationships
  .where("status", "published")
  .get();

// Posts with nested comments
const postWithComments = await db.posts().find(1).with(["comments.user"]); // Nested relationship

// Filter by related data
const postsByCategory = await db
  .posts()
  .whereHas("category", (query) => {
    query.where("name", "Technology");
  })
  .get();
```
