## Comparison Operators

### where() - Basic Conditions

The `where()` method is the foundation for filtering records:

```typescript
// Equality (default operator)
const activeUsers = await db.table("users").where("active", true).get();

const user = await db.table("users").where("email", "john@example.com").first();

// With specific operator
const adults = await db.table("users").where("age", ">=", 18).get();

const recentUsers = await db
  .table("users")
  .where("created_at", ">", "2023-01-01")
  .get();

// Inequality
const nonAdminUsers = await db
  .table("users")
  .where("role", "!=", "admin")
  .get();
```

**Supported Operators:**

- `=` (default), `!=`, `<>`, `>`, `>=`, `<`, `<=`
- `LIKE`, `NOT LIKE`, `ILIKE` (case insensitive, PostgreSQL)
- `IN`, `NOT IN`, `BETWEEN`, `NOT BETWEEN`
- `IS NULL`, `IS NOT NULL`

### Multiple WHERE Conditions (AND)

```typescript
// Multiple conditions with AND
const result = await db
  .table("users")
  .where("active", true)
  .where("age", ">=", 18)
  .where("city", "Madrid")
  .where("created_at", ">", "2023-01-01")
  .get();

// Equivalent to:
// SELECT * FROM users
// WHERE active = true
// AND age >= 18
// AND city = 'Madrid'
// AND created_at > '2023-01-01'
```

## Special Operators

### whereIn() - Multiple Values

```typescript
// Search by multiple IDs
const users = await db.table("users").whereIn("id", [1, 2, 3, 4, 5]).get();

// Search by multiple statuses
const posts = await db
  .table("posts")
  .whereIn("status", ["published", "featured", "archived"])
  .get();

// With subquery
const verifiedAuthors = db
  .table("authors")
  .select("id")
  .where("verified", true);

const posts = await db
  .table("posts")
  .whereIn("author_id", verifiedAuthors)
  .get();
```

### whereNotIn() - Exclude Values

```typescript
// Exclude users with specific statuses
const users = await db
  .table("users")
  .whereNotIn("status", ["banned", "deleted", "suspended"])
  .get();

// Exclude out-of-stock products
const availableProducts = await db
  .table("products")
  .whereNotIn("id", function () {
    this.select("product_id").from("products").where("stock", 0);
  })
  .get();
```

### whereBetween() - Value Ranges

```typescript
// Age between 18 and 65 years
const adults = await db.table("users").whereBetween("age", [18, 65]).get();

// Dates in a range
const recentPosts = await db
  .table("posts")
  .whereBetween("created_at", ["2023-01-01", "2023-12-31"])
  .get();

// Prices in a range
const midRangeProducts = await db
  .table("products")
  .whereBetween("price", [50, 200])
  .get();
```

### whereNotBetween() - Exclude Ranges

```typescript
// Products outside normal price range
const specialProducts = await db
  .table("products")
  .whereNotBetween("price", [10, 100])
  .get();

// Users outside typical age range
const atypicalUsers = await db
  .table("users")
  .whereNotBetween("age", [18, 65])
  .get();
```

## NULL Conditions

### whereNull() - Null Values

```typescript
// Users without email
const usersWithoutEmail = await db.table("users").whereNull("email").get();

// Posts without assigned category
const uncategorizedPosts = await db
  .table("posts")
  .whereNull("category_id")
  .get();

// Multiple null fields
const incompleteRecords = await db
  .table("users")
  .whereNull("phone")
  .whereNull("address")
  .get();
```

### whereNotNull() - Non-null Values

```typescript
// Users with email
const usersWithEmail = await db.table("users").whereNotNull("email").get();

// Posts with image
const postsWithImage = await db.table("posts").whereNotNull("image_url").get();

// Complete profile
const completeProfiles = await db
  .table("users")
  .whereNotNull("name")
  .whereNotNull("email")
  .whereNotNull("phone")
  .whereNotNull("address")
  .get();
```

## Text Search

### whereLike() - Pattern Matching

```typescript
// Names starting with 'John'
const johns = await db.table("users").where("name", "LIKE", "John%").get();

// Or using whereLike (more readable)
const johns = await db.table("users").whereLike("name", "John%").get();

// Emails ending in '.com'
const comEmails = await db.table("users").whereLike("email", "%.com").get();

// Names containing 'Garcia'
const garcias = await db.table("users").whereLike("name", "%Garcia%").get();

// Search with single character wildcard
const productCodes = await db
  .table("products")
  .whereLike("code", "PR_001") // _ represents a single character
  .get();
```

### whereILike() - Case Insensitive Search (PostgreSQL)

```typescript
// Case insensitive search
const users = await db
  .table("users")
  .whereILike("name", "%JOHN%") // Finds 'John', 'john', 'JOHN', etc.
  .get();
```

### whereNotLike() - Exclude Patterns

```typescript
// Users without Gmail emails
const nonGmail = await db
  .table("users")
  .whereNotLike("email", "%@gmail.com")
  .get();
```

## OR Conditions

### orWhere() - Alternative Conditions

```typescript
// Active OR VIP users
const specialUsers = await db
  .table("users")
  .where("active", true)
  .orWhere("vip", true)
  .get();

// Multiple OR conditions
const importantUsers = await db
  .table("users")
  .where("role", "admin")
  .orWhere("role", "moderator")
  .orWhere("vip", true)
  .get();

// Mix AND and OR with logical parentheses
const users = await db
  .table("users")
  .where("active", true)
  .where(function () {
    this.where("role", "premium").orWhere("points", ">", 1000);
  })
  .get();
// WHERE active = true AND (role = 'premium' OR points > 1000)
```

### Grouping OR Conditions

```typescript
// Complex OR conditions
const posts = await db
  .table("posts")
  .where("status", "published")
  .orWhere(function () {
    this.where("status", "draft").where("author_id", 1);
  })
  .get();
// WHERE status = 'published' OR (status = 'draft' AND author_id = 1)

// Multiple OR groups
const products = await db
  .table("products")
  .where(function () {
    this.where("category", "electronics").where("price", "<", 500);
  })
  .orWhere(function () {
    this.where("category", "books").where("discount", ">", 20);
  })
  .get();
```

## Advanced Conditions

### whereExists() - Existence of Related Records

```typescript
// Users who have published posts
const activeAuthors = await db
  .table("users")
  .whereExists(function () {
    this.select("*").from("posts").whereRaw("posts.user_id = users.id");
  })
  .get();

// Categories that have products
const categoriesWithProducts = await db
  .table("categories")
  .whereExists(function () {
    this.select("*")
      .from("products")
      .whereRaw("products.category_id = categories.id")
      .where("products.active", true);
  })
  .get();
```

### whereNotExists() - Absence of Related Records

```typescript
// Users who have never published
const inactiveUsers = await db
  .table("users")
  .whereNotExists(function () {
    this.select("*").from("posts").whereRaw("posts.user_id = users.id");
  })
  .get();

// Products without reviews
const productsWithoutReviews = await db
  .table("products")
  .whereNotExists(function () {
    this.select("*")
      .from("reviews")
      .whereRaw("reviews.product_id = products.id");
  })
  .get();
```

### whereRaw() - Custom SQL

```typescript
// Complex SQL conditions
const users = await db
  .table("users")
  .whereRaw("YEAR(birth_date) = ?", [1990])
  .get();

// Mathematical comparisons
const products = await db
  .table("products")
  .whereRaw("price * discount / 100 > ?", [50])
  .get();

// String functions
const users = await db
  .table("users")
  .whereRaw("CHAR_LENGTH(name) > ?", [10])
  .get();

// Date expressions
const orders = await db
  .table("orders")
  .whereRaw("DATE(created_at) = CURDATE()")
  .get();
```

## Practical Examples

### 1. Advanced User Search

```typescript
async function searchUsers(filters: {
  text?: string;
  ageMin?: number;
  ageMax?: number;
  cities?: string[];
  active?: boolean;
  vip?: boolean;
}) {
  let query = db.table("users");

  // Text search in name or email
  if (filters.text) {
    query = query.where(function () {
      this.whereLike("name", `%${filters.text}%`).orWhereLike(
        "email",
        `%${filters.text}%`
      );
    });
  }

  // Age filter
  if (filters.ageMin && filters.ageMax) {
    query = query.whereBetween("age", [filters.ageMin, filters.ageMax]);
  } else if (filters.ageMin) {
    query = query.where("age", ">=", filters.ageMin);
  } else if (filters.ageMax) {
    query = query.where("age", "<=", filters.ageMax);
  }

  // City filter
  if (filters.cities && filters.cities.length > 0) {
    query = query.whereIn("city", filters.cities);
  }

  // Active status
  if (filters.active !== undefined) {
    query = query.where("active", filters.active);
  }

  // VIP user
  if (filters.vip !== undefined) {
    query = query.where("vip", filters.vip);
  }

  return await query.get();
}
```

### 2. E-commerce Product Filters

```typescript
async function searchProducts(filters: {
  category?: number;
  priceMin?: number;
  priceMax?: number;
  brand?: string;
  available?: boolean;
  onSale?: boolean;
  minRating?: number;
}) {
  let query = db
    .table("products")
    .select([
      "products.*",
      "categories.name as category_name",
      "AVG(reviews.rating) as average_rating",
    ])
    .leftJoin("categories", "products.category_id", "=", "categories.id")
    .leftJoin("reviews", "products.id", "=", "reviews.product_id")
    .where("products.active", true);

  if (filters.category) {
    query = query.where("products.category_id", filters.category);
  }

  if (filters.priceMin && filters.priceMax) {
    query = query.whereBetween("products.price", [
      filters.priceMin,
      filters.priceMax,
    ]);
  }

  if (filters.brand) {
    query = query.whereLike("products.brand", `%${filters.brand}%`);
  }

  if (filters.available) {
    query = query.where("products.stock", ">", 0);
  }

  if (filters.onSale) {
    query = query.where("products.sale_price", "<", db.raw("products.price"));
  }

  query = query.groupBy("products.id");

  if (filters.minRating) {
    query = query.having("average_rating", ">=", filters.minRating);
  }

  return await query.get();
}
```

### 3. Admin Dashboard Statistics

```typescript
async function getAdminStats() {
  const [
    activeUsers,
    newUsersToday,
    publishedPosts,
    draftPosts,
    outOfStockProducts,
    pendingOrders,
  ] = await Promise.all([
    // Active users
    db.table("users").where("active", true).count(),

    // New users today
    db.table("users").whereRaw("DATE(created_at) = CURDATE()").count(),

    // Published posts
    db
      .table("posts")
      .where("status", "published")
      .whereNotNull("published_at")
      .count(),

    // Draft posts
    db.table("posts").where("status", "draft").count(),

    // Out of stock products
    db.table("products").where("active", true).where("stock", 0).count(),

    // Pending orders
    db
      .table("orders")
      .whereIn("status", ["pending", "processing"])
      .whereNotNull("order_date")
      .count(),
  ]);

  return {
    users: { active: activeUsers, newToday: newUsersToday },
    posts: { published: publishedPosts, draft: draftPosts },
    products: { outOfStock: outOfStockProducts },
    orders: { pending: pendingOrders },
  };
}
```

## Best Practices

### 1. Use Indexes Correctly

```typescript
// ✅ Good - uses indexes
const users = await db
  .table("users")
  .where("email", "john@example.com") // email should be indexed
  .where("active", true)
  .get();

// ❌ Avoid - functions in WHERE prevent indexes
const users = await db
  .table("users")
  .whereRaw("UPPER(email) = ?", ["JOHN@EXAMPLE.COM"])
  .get();
```

### 2. Validate User Input

```typescript
// Always validate and sanitize input
function validateEmail(email: string): boolean {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return emailRegex.test(email);
}

async function findUserByEmail(email: string) {
  if (!validateEmail(email)) {
    throw new Error("Invalid email");
  }

  return await db.table("users").where("email", email).first();
}
```

### 3. Use Parameters Instead of Concatenation

```typescript
// ✅ Good - prevents SQL injection
const users = await db
  .table("users")
  .where("name", "LIKE", `%${search}%`)
  .get();

// ❌ Avoid - vulnerable to SQL injection
const users = await db.raw(`
  SELECT * FROM users WHERE name LIKE '%${search}%'
`);
```
