## Basic Table Structure

::cube-code
---
code: |
    @database("myapp");

    @meta({
        name: "users";
        description: "System user accounts";
    });

    @columns({
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
            description: "Unique user identifier";
        };
        name: {
            type: "varchar";
            length: "100";
            options: ["not null"];
            description: "User's full name";
        };
        email: {
            type: "varchar";
            length: "255";
            options: ["not null", "unique"];
            description: "User email for authentication";
        };
        active: {
            type: "boolean";
            defaultValue: true;
            options: ["not null"];
            description: "User active/inactive status";
        };
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
        };
    });
---
::

## Supported Data Types

Dbcube supports 14 data types that work across all database engines:

### Text Types

::cube-code
---
code: |
    @columns({
        // Short text with specific length
        title: {
            type: "varchar";
            length: "255";
            description: "Article title";
        };

        // Alias for varchar (more familiar to developers)
        slug: {
            type: "string";
            length: "100";
            options: ["unique"];
        };

        // Long text without specific limit
        content: {
            type: "text";
            description: "Article full content";
        };
    });
---
::

### Numeric Types

::cube-code
---
code: |
    @columns({
        // Integers
        counter: {
            type: "int";
            options: ["unsigned"];
            defaultValue: 0;
        };

        // Decimals with specific precision
        price: {
            type: "decimal";
            length: "10,2";  // 10 total digits, 2 decimal places
            options: ["not null"];
        };

        // Floating point numbers
        rating: {
            type: "float";
            description: "Average rating";
        };

        // Double precision numbers
        latitude: {
            type: "double";
            description: "GPS latitude";
        };
    });
---
::

### Date and Time Types

::cube-code
---
code: |
    @columns({
        // Date only (YYYY-MM-DD)
        birth_date: {
            type: "date";
            description: "User birth date";
        };

        // Date and time without timezone
        event_datetime: {
            type: "datetime";
            description: "Event date and time";
        };

        // Timestamp with timezone
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
        };

        // Auto-updating timestamp
        updated_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP";
            options: ["not null"];
        };
    });
---
::

### Special Types

::cube-code
---
code: |

## Column Options

Options define characteristics and constraints for columns:

::cube-code
---
code: |
    @columns({
        example_column: {
            type: "varchar";
            length: "255";
            options: [
                "not null",      // Cannot be NULL
                "primary",       // Primary key
                "autoincrement", // Auto-increment (integers only)
                "unique",        // Unique constraint
                "index",         // Create index for fast searches
                "zerofill",      // Zero-fill numbers (MySQL)
                "required",      // Application-level required field
                "unsigned"       // Positive numbers only
            ];
            description: "Example with all available options";
        };
    });
---
::

### Common Combinations

::cube-code
---
code: |
    @columns({
        // Typical primary key ID
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };

        // Unique and required email
        email: {
            type: "varchar";
            length: "255";
            options: ["not null", "unique", "index"];
        };

        // Indexed column for searches
        category_id: {
            type: "int";
            options: ["not null", "index"];
        };

        // Positive number with default value
        points: {
            type: "int";
            options: ["unsigned", "not null"];
            defaultValue: 0;
        };
    });
---
::

## Default Values

Define values that are assigned automatically when no value is specified:

::cube-code
---
code: |
    @columns({
        // Literal values
        status: {
            type: "enum";
            enumValues: ["active", "inactive"];
            defaultValue: "active";
        };

        // Numeric values
        attempts: {
            type: "int";
            defaultValue: 0;
            options: ["unsigned"];
        };

        // Boolean values
        email_notifications: {
            type: "boolean";
            defaultValue: true;
        };

        // Date functions
        registration_date: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
        };

        // Empty JSON
        settings: {
            type: "json";
            defaultValue: {};
        };

        // Empty arrays (PostgreSQL)
        tags: {
            type: "json";
            defaultValue: [];
        };
    });
---

## Best Practices

### 1. Naming Conventions

::cube-code
---
code: |
    // ✅ Use snake_case for tables and columns
    @meta({
        name: "user_profiles";  // Not "UserProfiles" or "user-profiles"
    });

    @columns({
        user_id: { type: "int"; };           // Not "userId" or "user-id"
        birth_date: { type: "date"; };       // Not "birthDate"
        is_active: { type: "boolean"; };     // Not "isActive"
    });
---
::

### 2. IDs and Primary Keys

::cube-code
---
code: |
    @columns({
        // Always use 'id' as primary key
        id: {
            type: "int";
            options: ["primary", "autoincrement"];
        };

        // Foreign keys end with '_id'
        user_id: {
            type: "int";
            options: ["not null", "index"];
        };

        // UUIDs for distributed systems
        uuid: {
            type: "varchar";
            length: "36";
            options: ["unique"];
        };
    });
---
::

### 3. Standard Timestamps

::cube-code
---
code: |
    @columns({
        // Always include timestamps
        created_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP";
            options: ["not null"];
        };

        updated_at: {
            type: "timestamp";
            defaultValue: "CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP";
            options: ["not null"];
        };

        // For soft deletes
        deleted_at: {
            type: "timestamp";
            description: "NULL if not deleted";
        };
    });
---
::

## Common Validations

### Common Errors and Solutions

::cube-code
---
code: |
    // ❌ VARCHAR without length
    name: {
        type: "varchar";  // Error: missing 'length'
        options: ["not null"];
    };

    // ✅ Correct VARCHAR definition
    name: {
        type: "varchar";
        length: "255";
        options: ["not null"];
    };

    // ❌ autoincrement on non-integer type
    code: {
        type: "varchar";
        options: ["autoincrement"];  // Error: only for int
    };

    // ✅ autoincrement only on integers
    id: {
        type: "int";
        options: ["primary", "autoincrement"];
    };

    // ❌ Enum without values
    status: {
        type: "enum";  // Error: missing 'enumValues'
        options: ["not null"];
    };

    // ✅ Enum with defined values
    status: {
        type: "enum";
        enumValues: ["active", "inactive"];
        options: ["not null"];
    };
---
