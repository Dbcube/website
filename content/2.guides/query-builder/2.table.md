## Basic Usage

```typescript
import { Dbcube } from "dbcube";

const dbcube = new Dbcube();
await dbcube.init();
const db = dbcube.database("myapp");

// Get table instance
const usersTable = db.table("users");
```

## Selection Methods

### select()

Specify which columns to retrieve from the table.

```typescript
// Select all columns
const users = await db.table("users").select().get();

// Select specific columns
const users = await db.table("users").select(["id", "name", "email"]).get();

// Select with aliases
const users = await db
  .table("users")
  .select([
    "name as full_name",
    "email as email_address",
    "created_at as registration_date",
  ])
  .get();

// Select with expressions
const users = await db
  .table("users")
  .select([
    "name",
    "UPPER(email) as email_upper",
    "YEAR(created_at) as registration_year",
  ])
  .get();
```

**Parameters:**

- `columns` (string[] | string, optional): Column names, expressions, or aliases

**Returns:** `Table` instance for method chaining

### distinct()

Select only unique rows.

```typescript
// Get unique user statuses
const statuses = await db.table("users").select("status").distinct().get();

// Get unique combinations
const combinations = await db
  .table("posts")
  .select(["author_id", "category_id"])
  .distinct()
  .get();
```

**Returns:** `Table` instance for method chaining

## Filtering Methods

### where()

Add WHERE conditions to filter results.

```typescript
// Simple equality
const activeUsers = await db.table("users").where("active", true).get();

// With operator
const adults = await db.table("users").where("age", ">=", 18).get();

// Multiple conditions (AND)
const results = await db
  .table("users")
  .where("active", true)
  .where("age", ">=", 18)
  .where("city", "New York")
  .get();

// Different operators
const users = await db
  .table("users")
  .where("name", "LIKE", "%John%")
  .where("created_at", ">", "2023-01-01")
  .where("status", "!=", "banned")
  .get();
```

**Supported Operators:**

- `=` (default), `!=`, `<>`, `>`, `>=`, `<`, `<=`
- `LIKE`, `NOT LIKE`, `ILIKE` (case-insensitive, PostgreSQL)
- `IN`, `NOT IN`, `BETWEEN`, `NOT BETWEEN`
- `IS NULL`, `IS NOT NULL`

### whereIn()

Filter where column value is in a list of values.

```typescript
const users = await db.table("users").whereIn("id", [1, 2, 3, 4, 5]).get();
const posts = await db
  .table("posts")
  .whereIn("status", ["published", "featured"])
  .get();

// With subquery
const authorIds = db.table("authors").select("id").where("verified", true);
const posts = await db.table("posts").whereIn("author_id", authorIds).get();
```

### whereNotIn()

Filter where column value is NOT in a list of values.

```typescript
const users = await db
  .table("users")
  .whereNotIn("status", ["banned", "deleted"])
  .get();
```

### whereBetween()

Filter where column value is between two values (inclusive).

```typescript
const adults = await db.table("users").whereBetween("age", [18, 65]).get();
const recentPosts = await db
  .table("posts")
  .whereBetween("created_at", ["2023-01-01", "2023-12-31"])
  .get();
```

### whereNotBetween()

Filter where column value is NOT between two values.

```typescript
const outliers = await db
  .table("products")
  .whereNotBetween("price", [10, 100])
  .get();
```

### whereNull() / whereNotNull()

Filter for NULL or NOT NULL values.

```typescript
// Users without email
const usersWithoutEmail = await db.table("users").whereNull("email").get();

// Users with email
const usersWithEmail = await db.table("users").whereNotNull("email").get();
```

### whereLike()

Pattern matching with LIKE operator.

```typescript
// Names starting with 'John'
const johns = await db.table("users").whereLike("name", "John%").get();

// Emails ending with '.com'
const comUsers = await db.table("users").whereLike("email", "%.com").get();

// Names containing 'smith' (case-insensitive)
const smiths = await db.table("users").whereILike("name", "%smith%").get();
```

### orWhere()

Add OR conditions to the query.

```typescript
const users = await db
  .table("users")
  .where("active", true)
  .orWhere("vip", true)
  .get();

// Complex OR conditions
const posts = await db
  .table("posts")
  .where("status", "published")
  .orWhere(function () {
    this.where("status", "draft").where("author_id", 1);
  })
  .get();
```

### whereExists() / whereNotExists()

Filter based on existence of related records.

```typescript
// Users who have posted
const activeAuthors = await db
  .table("users")
  .whereExists(function () {
    this.select("*").from("posts").whereRaw("posts.user_id = users.id");
  })
  .get();

// Users who haven't posted
const inactiveUsers = await db
  .table("users")
  .whereNotExists(function () {
    this.select("*").from("posts").whereRaw("posts.user_id = users.id");
  })
  .get();
```

## Grouping and Having

### groupBy()

Group results by one or more columns.

```typescript
// Group by single column
const statusCounts = await db
  .table("users")
  .select(["status", db.raw("COUNT(*) as count")])
  .groupBy("status")
  .get();

// Group by multiple columns
const monthlyCounts = await db
  .table("posts")
  .select([
    db.raw("YEAR(created_at) as year"),
    db.raw("MONTH(created_at) as month"),
    db.raw("COUNT(*) as count"),
  ])
  .groupBy(["year", "month"])
  .get();
```

### having()

Filter grouped results (like WHERE but for groups).

```typescript
// Categories with more than 5 posts
const popularCategories = await db
  .table("posts")
  .select(["category_id", db.raw("COUNT(*) as post_count")])
  .groupBy("category_id")
  .having("post_count", ">", 5)
  .get();

// Authors with average rating above 4
const topAuthors = await db
  .table("reviews")
  .select(["author_id", db.raw("AVG(rating) as avg_rating")])
  .groupBy("author_id")
  .having("avg_rating", ">", 4)
  .get();
```

## Ordering and Limiting

### orderBy()

Sort results by one or more columns.

```typescript
// Single column ascending (default)
const users = await db.table("users").orderBy("name").get();

// Single column descending
const users = await db.table("users").orderBy("created_at", "desc").get();

// Multiple columns
const users = await db
  .table("users")
  .orderBy("status", "asc")
  .orderBy("created_at", "desc")
  .orderBy("name", "asc")
  .get();

// Order by expression
const posts = await db
  .table("posts")
  .orderBy(db.raw("CHAR_LENGTH(title)"), "desc")
  .get();
```

### orderByRaw()

Order by raw SQL expressions.

```typescript
const products = await db
  .table("products")
  .orderByRaw("price * discount_percentage DESC")
  .get();
```

### limit()

Limit the number of results returned.

```typescript
const latestPosts = await db
  .table("posts")
  .orderBy("created_at", "desc")
  .limit(10)
  .get();
```

### offset()

Skip a number of results (for pagination).

```typescript
// Page 2, 10 items per page
const page2Users = await db
  .table("users")
  .orderBy("id")
  .limit(10)
  .offset(10)
  .get();

// Helper for pagination
function paginate(page, perPage) {
  return {
    limit: perPage,
    offset: (page - 1) * perPage,
  };
}

const { limit, offset } = paginate(3, 20);
const page3Users = await db.table("users").limit(limit).offset(offset).get();
```

## Joins

### join()

Perform inner joins with other tables.

```typescript
// Basic join
const usersWithPosts = await db
  .table("users")
  .join("posts", "users.id", "=", "posts.user_id")
  .select(["users.name", "posts.title"])
  .get();

// Multiple joins
const fullData = await db
  .table("posts")
  .join("users", "posts.user_id", "=", "users.id")
  .join("categories", "posts.category_id", "=", "categories.id")
  .select([
    "posts.title",
    "users.name as author",
    "categories.name as category",
  ])
  .get();

// Join with conditions
const results = await db
  .table("users")
  .join("posts", function () {
    this.on("users.id", "=", "posts.user_id").andOn(
      "posts.status",
      "=",
      db.raw("?", ["published"])
    );
  })
  .get();
```

### leftJoin() / rightJoin()

Perform outer joins.

```typescript
// Left join (includes all users, even without posts)
const allUsersWithPosts = await db
  .table("users")
  .leftJoin("posts", "users.id", "=", "posts.user_id")
  .select([
    "users.name",
    "posts.title",
    db.raw("COUNT(posts.id) as post_count"),
  ])
  .groupBy("users.id")
  .get();

// Right join
const allPostsWithUsers = await db
  .table("users")
  .rightJoin("posts", "users.id", "=", "posts.user_id")
  .select(["users.name", "posts.title"])
  .get();
```

### crossJoin()

Perform cross joins (Cartesian product).

```typescript
const combinations = await db
  .table("colors")
  .crossJoin("sizes")
  .select(["colors.name as color", "sizes.name as size"])
  .get();
```

## Aggregation Functions

### count()

Count the number of matching records.

```typescript
// Total users
const totalUsers = await db.table("users").count();

// Active users
const activeUsers = await db.table("users").where("active", true).count();

// Count specific column
const usersWithEmail = await db.table("users").count("email");

// Count with alias
const counts = await db
  .table("posts")
  .select("category_id")
  .count("id as post_count")
  .groupBy("category_id")
  .get();
```

### sum() / avg() / min() / max()

Aggregate functions for numeric data.

```typescript
// Sum of all product prices
const totalValue = await db.table("products").sum("price");

// Average user age
const avgAge = await db.table("users").avg("age");

// Min/Max prices
const priceRange = await db
  .table("products")
  .select([db.min("price as min_price"), db.max("price as max_price")])
  .first();

// Grouped aggregations
const categoryStats = await db
  .table("products")
  .select([
    "category_id",
    db.count("* as product_count"),
    db.avg("price as avg_price"),
    db.sum("inventory as total_inventory"),
  ])
  .groupBy("category_id")
  .get();
```

## Data Modification

### insert()

Insert new records into the table.

```typescript
// Insert single record
const user = await db.table("users").insert({
  name: "John Doe",
  email: "john@example.com",
  active: true,
});

// Insert multiple records
const users = await db.table("users").insert([
  { name: "Alice", email: "alice@example.com" },
  { name: "Bob", email: "bob@example.com" },
  { name: "Carol", email: "carol@example.com" },
]);

// Insert with selected columns
const post = await db.table("posts").insert({
  title: "New Post",
  content: "Post content...",
  user_id: 1,
  created_at: new Date(),
});
```

### update()

Update existing records.

```typescript
// Update single record by ID
const result = await db.table("users").where("id", 1).update({
  name: "Jane Doe",
  updated_at: new Date(),
});

// Update multiple records
const result = await db
  .table("posts")
  .where("status", "draft")
  .where("user_id", 1)
  .update({
    status: "published",
    published_at: new Date(),
  });

// Update with expressions
const result = await db
  .table("products")
  .where("category_id", 1)
  .update({
    price: db.raw("price * 1.1"), // 10% increase
    updated_at: db.raw("NOW()"),
  });
```

### upsert()

Insert or update records (if they exist).

```typescript
// Upsert single record
const user = await db.table("users").upsert(
  {
    email: "john@example.com",
    name: "John Doe",
    active: true,
  },
  ["email"]
); // Conflict target

// Upsert multiple records
const users = await db.table("users").upsert(
  [
    { email: "alice@example.com", name: "Alice Smith" },
    { email: "bob@example.com", name: "Bob Jones" },
  ],
  ["email"]
);
```

### delete()

Delete records from the table.

```typescript
// Delete specific record
const result = await db.table("users").where("id", 1).delete();

// Delete multiple records
const result = await db
  .table("posts")
  .where("status", "draft")
  .where("created_at", "<", "2022-01-01")
  .delete();

// Soft delete (update instead of delete)
const result = await db
  .table("users")
  .where("id", 1)
  .update({ deleted_at: new Date() });
```

## Advanced Queries

### subqueries()

Use subqueries in various contexts.

```typescript
// Subquery in WHERE
const popularPosts = await db
  .table("posts")
  .where("view_count", ">", function () {
    this.select(db.raw("AVG(view_count)")).from("posts");
  })
  .get();

// Subquery in SELECT
const usersWithPostCount = await db
  .table("users")
  .select([
    "users.*",
    db
      .table("posts")
      .count("*")
      .where("user_id", db.raw("users.id"))
      .as("post_count"),
  ])
  .get();

// Subquery in FROM
const topCategories = await db
  .from(function () {
    this.select(["category_id", db.count("* as post_count")])
      .from("posts")
      .groupBy("category_id")
      .as("category_stats");
  })
  .where("post_count", ">", 10)
  .get();
```

### unions()

Combine results from multiple queries.

```typescript
// Simple union
const allContent = await db
  .table("posts")
  .select(["title", "created_at", db.raw('"post" as type')])
  .union(function () {
    this.select([
      "name as title",
      "created_at",
      db.raw('"category" as type'),
    ]).from("categories");
  })
  .orderBy("created_at", "desc")
  .get();

// Union all (includes duplicates)
const allEmails = await db
  .table("users")
  .select("email")
  .unionAll(function () {
    this.select("email").from("subscribers");
  })
  .get();
```

### with() (CTEs)

Use Common Table Expressions for complex queries.

```typescript
// Recursive CTE example
const hierarchicalData = await db
  .with("categories_tree", function () {
    this.select(["id", "name", "parent_id", db.raw("1 as level")])
      .from("categories")
      .where("parent_id", null)
      .unionAll(function () {
        this.select(["c.id", "c.name", "c.parent_id", db.raw("ct.level + 1")])
          .from("categories as c")
          .join("categories_tree as ct", "c.parent_id", "=", "ct.id");
      });
  })
  .select("*")
  .from("categories_tree")
  .orderBy(["level", "name"])
  .get();
```

## Raw SQL

### raw()

Execute raw SQL expressions.

```typescript
// Raw SELECT
const users = await db
  .table("users")
  .select(db.raw('CONCAT(first_name, " ", last_name) as full_name'))
  .get();

// Raw WHERE
const users = await db
  .table("users")
  .whereRaw("YEAR(created_at) = ?", [2023])
  .get();

// Raw JOIN
const results = await db
  .table("users")
  .joinRaw("LEFT JOIN posts ON users.id = posts.user_id AND posts.status = ?", [
    "published",
  ])
  .get();

// Raw ORDER BY
const products = await db
  .table("products")
  .orderByRaw('FIELD(category, "electronics", "books", "clothing")')
  .get();
```

## Utility Methods

### clone()

Clone a query builder to reuse base queries.

```typescript
const baseQuery = db
  .table("users")
  .select(["id", "name", "email"])
  .where("active", true);

// Use the base query for different purposes
const recentUsers = baseQuery
  .clone()
  .where("created_at", ">", "2023-01-01")
  .get();

const vipUsers = baseQuery.clone().where("vip", true).get();
```

### toSql()

Get the SQL query string without executing.

```typescript
const query = db
  .table("users")
  .select(["name", "email"])
  .where("active", true)
  .orderBy("created_at", "desc")
  .limit(10);

console.log(query.toSql());
// Output: SELECT name, email FROM users WHERE active = ? ORDER BY created_at DESC LIMIT ?
```

### explain()

Get query execution plan.

```typescript
const explanation = await db
  .table("users")
  .select(["name", "email"])
  .where("active", true)
  .explain();

console.log(explanation);
```

## Performance Optimization

### Indexing Hints

```typescript
// Use index hints (MySQL)
const users = await db
  .table("users")
  .useIndex(["idx_email", "idx_created_at"])
  .where("email", "LIKE", "%@example.com")
  .get();

// Force index
const posts = await db
  .table("posts")
  .forceIndex("idx_status_created")
  .where("status", "published")
  .orderBy("created_at", "desc")
  .get();
```

### Query Optimization

```typescript
// Efficient pagination with cursor
async function cursorPaginate(lastId = 0, limit = 20) {
  return await db
    .table("posts")
    .where("id", ">", lastId)
    .orderBy("id")
    .limit(limit)
    .get();
}

// Batch operations
async function batchUpdate(updates) {
  const promises = updates.map(({ id, data }) =>
    db.table("users").where("id", id).update(data)
  );

  return await Promise.all(promises);
}
```

## Error Handling

```typescript
try {
  const users = await db.table("users").where("active", true).get();
} catch (error) {
  if (error.code === "ER_NO_SUCH_TABLE") {
    console.error("Table does not exist");
  } else if (error.code === "ER_BAD_FIELD_ERROR") {
    console.error("Invalid column name");
  } else {
    console.error("Query error:", error.message);
  }
}
```
