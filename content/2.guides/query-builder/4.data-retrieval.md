## Query Execution Methods

### get() - Retrieve Multiple Records

The `get()` method executes the query and returns all records that match the criteria:

```typescript
// Get all active users
const activeUsers = await db.table("users").where("active", true).get();

// Get published posts with specific columns
const posts = await db
  .table("posts")
  .select(["id", "title", "created_at"])
  .where("status", "published")
  .orderBy("created_at", "desc")
  .get();

// Result with joins
const postsWithAuthor = await db
  .table("posts")
  .select(["posts.title", "users.name as author"])
  .join("users", "posts.user_id", "=", "users.id")
  .get();

console.log(`Found ${posts.length} posts`);
```

**Returns:** `Promise<DatabaseRecord[]>` - Array of result objects

### first() - Get the First Record

The `first()` method retrieves only the first record that matches the criteria:

```typescript
// Get the first active user
const user = await db.table("users").where("active", true).first();

if (user) {
  console.log(`User found: ${user.name}`);
} else {
  console.log("No active user found");
}

// Get the most recent post
const recentPost = await db
  .table("posts")
  .where("status", "published")
  .orderBy("created_at", "desc")
  .first();

// First user from a specific city
const madridUser = await db
  .table("users")
  .select(["id", "name", "email"])
  .where("city", "Madrid")
  .where("active", true)
  .first();
```

**Returns:** `Promise<DatabaseRecord | null>` - A result object or null if not found

### find() - Search by Primary Key

The `find()` method is a quick way to search for a record by its primary key (usually `id`):

```typescript
// Find user by ID
const user = await db.table("users").find(1);

if (user) {
  console.log(`User ID 1: ${user.name}`);
} else {
  console.log("User not found");
}

// Find multiple records by ID
const users = await Promise.all([
  db.table("users").find(1),
  db.table("users").find(2),
  db.table("users").find(3),
]);

// Filter found users
const foundUsers = users.filter((u) => u !== null);
```

**Parameters:**

- `id` (any): Primary key value

**Returns:** `Promise<DatabaseRecord | null>` - A result object or null if not found

### findMany() - Find Multiple by IDs

```typescript
// Find multiple users by their IDs
const users = await db.table("users").whereIn("id", [1, 2, 3, 4, 5]).get();

// Or use a custom helper function
async function findMany(
  table: string,
  ids: number[]
): Promise<DatabaseRecord[]> {
  return await db.table(table).whereIn("id", ids).get();
}

const products = await findMany("products", [10, 20, 30]);
```

## Count and Aggregation Methods

### count() - Count Records

```typescript
// Count total users
const totalUsers = await db.table("users").count();

// Count active users
const activeUsers = await db.table("users").where("active", true).count();

// Count with alias
const result = await db
  .table("posts")
  .select("category_id")
  .count("id as total_posts")
  .groupBy("category_id")
  .get();

// Count non-null records
const usersWithPhone = await db.table("users").count("phone"); // Only counts records where phone is NOT NULL

console.log(`There are ${totalUsers} registered users`);
console.log(`${activeUsers} users are active`);
```

**Returns:** `Promise<number>` - Number of records

### exists() - Check Existence

```typescript
// Check if a user exists with specific email
const emailExists = await db
  .table("users")
  .where("email", "john@example.com")
  .exists();

if (emailExists) {
  console.log("Email already registered");
} else {
  console.log("Email is available");
}

// Check if there are posts published today
const hasPostsToday = await db
  .table("posts")
  .where("status", "published")
  .whereRaw("DATE(created_at) = CURDATE()")
  .exists();

// Check existence with joins
const hasPermission = await db
  .table("users")
  .join("roles", "users.role_id", "=", "roles.id")
  .where("users.id", userId)
  .where("roles.name", "admin")
  .exists();
```

**Returns:** `Promise<boolean>` - true if at least one record exists

### Aggregation Functions

```typescript
// Sum total
const totalSales = await db
  .table("orders")
  .where("status", "completed")
  .sum("total");

// Average
const averageAge = await db.table("users").where("active", true).avg("age");

// Minimum and maximum values
const [minPrice, maxPrice] = await Promise.all([
  db.table("products").min("price"),
  db.table("products").max("price"),
]);

// Multiple aggregations in one query
const productStats = await db
  .table("products")
  .select([
    "category_id",
    db.raw("COUNT(*) as total_products"),
    db.raw("AVG(price) as average_price"),
    db.raw("SUM(stock) as total_stock"),
    db.raw("MIN(price) as min_price"),
    db.raw("MAX(price) as max_price"),
  ])
  .where("active", true)
  .groupBy("category_id")
  .get();

console.log(`Total sales: $${totalSales}`);
console.log(`Average age: ${Math.round(averageAge)} years`);
```

## Pagination and Limiting

### limit() - Limit Results

```typescript
// Get the latest 10 posts
const latestPosts = await db
  .table("posts")
  .orderBy("created_at", "desc")
  .limit(10)
  .get();

// Top 5 best-selling products
const topProducts = await db
  .table("products")
  .select(["name", "total_sales"])
  .orderBy("total_sales", "desc")
  .limit(5)
  .get();
```

### offset() - Skip Records

```typescript
// Manual pagination
const page = 2;
const perPage = 20;
const offset = (page - 1) * perPage;

const users = await db
  .table("users")
  .select(["id", "name", "email"])
  .orderBy("name")
  .limit(perPage)
  .offset(offset)
  .get();
```

### Complete Pagination Function

```typescript
interface PaginationResult<T> {
  data: T[];
  pagination: {
    page: number;
    perPage: number;
    total: number;
    totalPages: number;
    hasNext: boolean;
    hasPrev: boolean;
  };
}

async function paginate<T>(
  query: any,
  page: number = 1,
  perPage: number = 20
): Promise<PaginationResult<T>> {
  const offset = (page - 1) * perPage;

  // Clone query for counting
  const countQuery = query.clone();

  // Execute both queries in parallel
  const [data, total] = await Promise.all([
    query.limit(perPage).offset(offset).get(),
    countQuery.count(),
  ]);

  const totalPages = Math.ceil(total / perPage);

  return {
    data,
    pagination: {
      page,
      perPage,
      total,
      totalPages,
      hasNext: page < totalPages,
      hasPrev: page > 1,
    },
  };
}

// Usage of pagination function
const paginatedResult = await paginate(
  db
    .table("users")
    .select(["id", "name", "email"])
    .where("active", true)
    .orderBy("name"),
  2, // page 2
  15 // 15 per page
);

console.log(
  `Page ${paginatedResult.pagination.page} of ${paginatedResult.pagination.totalPages}`
);
console.log(`Total records: ${paginatedResult.pagination.total}`);
```

## Processing Large Datasets

### chunk() - Process in Batches

To handle large amounts of data without overloading memory:

```typescript
// Process all users in batches of 100
await db
  .table("users")
  .where("active", true)
  .chunk(100, async (users) => {
    console.log(`Processing batch of ${users.length} users`);

    for (const user of users) {
      // Process each user
      await processUser(user);
    }
  });

// Example: Update emails in batches
await db
  .table("users")
  .where("email_validated", false)
  .chunk(50, async (users) => {
    const promises = users.map(async (user) => {
      const normalizedEmail = user.email.toLowerCase();
      return db
        .table("users")
        .where("id", user.id)
        .update({ email: normalizedEmail });
    });

    await Promise.all(promises);
    console.log(`Updated ${users.length} emails`);
  });
```

### stream() - Stream Processing

```typescript
// Process records one by one (if supported by driver)
const stream = db.table("logs").where("date", ">", "2023-01-01").stream();

stream.on("data", (record) => {
  // Process each record individually
  processLog(record);
});

stream.on("end", () => {
  console.log("Log processing completed");
});
```

## Practical Examples

### 1. Dashboard Statistics

```typescript
async function getDashboardStats() {
  const [
    totalUsers,
    activeUsers,
    newUsers,
    totalPosts,
    publishedPosts,
    todaySales,
    monthlySales,
    topProduct,
  ] = await Promise.all([
    // Total registered users
    db.table("users").count(),

    // Active users
    db.table("users").where("active", true).count(),

    // New users in last 7 days
    db
      .table("users")
      .where("created_at", ">", db.raw("NOW() - INTERVAL 7 DAY"))
      .count(),

    // Total posts
    db.table("posts").count(),

    // Published posts
    db.table("posts").where("status", "published").count(),

    // Today's sales
    db
      .table("orders")
      .where("status", "completed")
      .whereRaw("DATE(created_at) = CURDATE()")
      .sum("total"),

    // Monthly sales
    db
      .table("orders")
      .where("status", "completed")
      .whereRaw("MONTH(created_at) = MONTH(NOW())")
      .whereRaw("YEAR(created_at) = YEAR(NOW())")
      .sum("total"),

    // Top selling product
    db
      .table("products")
      .select(["name", "total_sales"])
      .orderBy("total_sales", "desc")
      .first(),
  ]);

  return {
    users: {
      total: totalUsers,
      active: activeUsers,
      new: newUsers,
      activityRate: ((activeUsers / totalUsers) * 100).toFixed(1),
    },
    content: {
      totalPosts: totalPosts,
      published: publishedPosts,
      publishRate: ((publishedPosts / totalPosts) * 100).toFixed(1),
    },
    sales: {
      today: todaySales || 0,
      month: monthlySales || 0,
      topProduct: topProduct?.name || "N/A",
    },
  };
}
```

### 2. Autocomplete Search System

```typescript
async function getAutocomplete(term: string, limit: number = 10) {
  if (!term || term.length < 2) {
    return [];
  }

  const [users, products, categories] = await Promise.all([
    // Search users
    db
      .table("users")
      .select(["id", "name", "email", db.raw('"user" as type')])
      .where("active", true)
      .where(function () {
        this.whereLike("name", `%${term}%`).orWhereLike("email", `%${term}%`);
      })
      .limit(limit)
      .get(),

    // Search products
    db
      .table("products")
      .select(["id", "name", "description", db.raw('"product" as type')])
      .where("active", true)
      .where(function () {
        this.whereLike("name", `%${term}%`).orWhereLike(
          "description",
          `%${term}%`
        );
      })
      .limit(limit)
      .get(),

    // Search categories
    db
      .table("categories")
      .select(["id", "name", "description", db.raw('"category" as type')])
      .whereLike("name", `%${term}%`)
      .limit(limit)
      .get(),
  ]);

  return [...users, ...products, ...categories]
    .sort((a, b) => a.name.localeCompare(b.name))
    .slice(0, limit);
}
```

### 3. Reporting System

```typescript
async function generateSalesReport(startDate: string, endDate: string) {
  const baseQuery = db
    .table("orders")
    .whereBetween("created_at", [startDate, endDate])
    .where("status", "completed");

  const [summary, dailySales, topProducts, topCustomers] = await Promise.all([
    // General summary
    baseQuery
      .clone()
      .select([
        db.raw("COUNT(*) as total_orders"),
        db.raw("SUM(total) as total_sales"),
        db.raw("AVG(total) as average_sale"),
        db.raw("MIN(total) as min_sale"),
        db.raw("MAX(total) as max_sale"),
      ])
      .first(),

    // Daily sales
    baseQuery
      .clone()
      .select([
        db.raw("DATE(created_at) as date"),
        db.raw("COUNT(*) as orders"),
        db.raw("SUM(total) as sales"),
      ])
      .groupBy(db.raw("DATE(created_at)"))
      .orderBy(db.raw("DATE(created_at)"))
      .get(),

    // Top selling products
    db
      .table("order_items")
      .select([
        "products.name",
        db.raw("SUM(order_items.quantity) as quantity_sold"),
        db.raw("SUM(order_items.subtotal) as revenue"),
      ])
      .join("products", "order_items.product_id", "=", "products.id")
      .join("orders", "order_items.order_id", "=", "orders.id")
      .whereBetween("orders.created_at", [startDate, endDate])
      .where("orders.status", "completed")
      .groupBy("products.id")
      .orderBy("quantity_sold", "desc")
      .limit(10)
      .get(),

    // Top customers
    baseQuery
      .clone()
      .select([
        "users.name",
        "users.email",
        db.raw("COUNT(*) as total_orders"),
        db.raw("SUM(orders.total) as total_spent"),
      ])
      .join("users", "orders.user_id", "=", "users.id")
      .groupBy("users.id")
      .orderBy("total_spent", "desc")
      .limit(10)
      .get(),
  ]);

  return {
    period: { start: startDate, end: endDate },
    summary: summary,
    dailySales: dailySales,
    topProducts: topProducts,
    topCustomers: topCustomers,
  };
}
```

## Optimization and Best Practices

### 1. Select Only Necessary Columns

```typescript
// ✅ Good - selects only necessary columns
const users = await db.table("users").select(["id", "name", "email"]).get();

// ❌ Avoid - loads all columns unnecessarily
const users = await db.table("users").get();
```

### 2. Use Count Instead of get().length

```typescript
// ✅ Good - more efficient for counting
const totalUsers = await db.table("users").count();

// ❌ Avoid - loads all records just to count
const users = await db.table("users").get();
const total = users.length;
```

### 3. Implement Cache for Frequent Queries

```typescript
// Simple in-memory cache
const cache = new Map();
const CACHE_TTL = 5 * 60 * 1000; // 5 minutes

async function getActiveUsersWithCache() {
  const cacheKey = "active_users";
  const cached = cache.get(cacheKey);

  if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
    return cached.data;
  }

  const users = await db
    .table("users")
    .select(["id", "name", "email"])
    .where("active", true)
    .get();

  cache.set(cacheKey, {
    data: users,
    timestamp: Date.now(),
  });

  return users;
}
```
