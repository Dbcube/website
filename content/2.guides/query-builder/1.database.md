## Key Features

- **Fluent Interface**: Chainable methods for building complex queries
- **Multi-Engine Support**: Compatible with MySQL, PostgreSQL, SQLite and more
- **Type-Safe**: Full TypeScript support with comprehensive typing
- **Transactions**: Complete database transaction support
- **Query Builder**: Intuitive and powerful query construction
- **Raw SQL**: Ability to execute custom SQL when needed

## Initial Setup

### Installation and Initialization

```typescript
import { Dbcube } from "dbcube";

// Create Dbcube instance
const dbcube = new Dbcube();

// Initialize connection
await dbcube.init();

// Get database instance
const db = dbcube.database("myapp");
```

### Connection Configuration

```typescript
// dbcube.config.js or dbcube.config.ts
export default {
  databases: {
    myapp: {
      driver: "mysql",
      host: "localhost",
      port: 3306,
      database: "my_application",
      username: "user",
      password: "password",
      charset: "utf8mb4",
      collation: "utf8mb4_unicode_ci",
    },
  },
};
```

## Core Concepts

### Table Selection

All operations begin by selecting a table:

```typescript
// Get a table instance
const users = db.table("users");
const posts = db.table("posts");
const categories = db.table("categories");
```

### Method Chaining

Dbcube uses a fluent pattern that allows method chaining:

```typescript
const result = await db
  .table("users")
  .select(["id", "name", "email"])
  .where("active", true)
  .where("age", ">=", 18)
  .orderBy("name")
  .limit(10)
  .get();
```

### Return Types

```typescript
// Type definitions
export interface DatabaseRecord {
  [key: string]: any;
}

// Multiple records
const users: DatabaseRecord[] = await db.table("users").get();

// Single record
const user: DatabaseRecord | null = await db.table("users").first();

// Count
const total: number = await db.table("users").count();
```

## Basic Examples

### Simple Query

```typescript
// Get all active users
const activeUsers = await db.table("users").where("active", true).get();

console.log(`Found ${activeUsers.length} active users`);
```

### Query with Column Selection

```typescript
// Select specific columns
const profiles = await db
  .table("users")
  .select(["id", "name", "email", "created_at"])
  .where("verified", true)
  .orderBy("name")
  .get();
```

### Search by ID

```typescript
// Find user by ID
const user = await db.table("users").find(1);

if (user) {
  console.log(`User found: ${user.name}`);

  console.log("User not found");

```

## Error Handling

```typescript
try {
  const users = await db.table("users").where("active", true).get();

  console.log("Users retrieved:", users);
} catch (error) {
  if (error.code === "ER_NO_SUCH_TABLE") {
    console.error("Table does not exist");
  } else if (error.code === "ER_BAD_FIELD_ERROR") {
    console.error("Invalid column name");
  } else {
    console.error("Database error:", error.message);
  }
}
```

## Best Practices

### 1. Always Use Try-Catch

```typescript
async function getUsers() {
  try {
    return await db.table("users").get();
  } catch (error) {
    console.error("Error fetching users:", error);
    throw error;
  }
}
```

### 2. Select Specific Columns

```typescript
// ✅ Good - select only needed columns
const users = await db.table("users").select(["id", "name", "email"]).get();

// ❌ Avoid - selects all columns unnecessarily
const users = await db.table("users").get();
```

### 3. Use Pagination for Large Datasets

```typescript
// Implement pagination
async function getPaginatedUsers(page = 1, perPage = 20) {
  const offset = (page - 1) * perPage;

  const [users, total] = await Promise.all([
    db
      .table("users")
      .select(["id", "name", "email"])
      .orderBy("name")
      .limit(perPage)
      .offset(offset)
      .get(),
    db.table("users").count(),
  ]);

  return {
    data: users,
    pagination: {
      page,
      perPage,
      total,
      totalPages: Math.ceil(total / perPage),
    },
  };
}
```

### 4. Validate Input Data

```typescript
async function findUserByEmail(email: string) {
  // Validate input
  if (!email || typeof email !== "string") {
    throw new Error("Invalid email");
  }

  return await db.table("users").where("email", email).first();
}
```

## Query Methods

### select()

Specify columns to select from the table.

```typescript
// Select all columns
const allUsers = await db.table("users").select().get();

// Select specific columns
const userNames = await db.table("users").select(["id", "name", "email"]).get();

// Select with alias
const users = await db
  .table("users")
  .select(["name as full_name", "email"])
  .get();
```

**Parameters:**

- `columns` (string[] | string, optional): Array of column names or single column. Defaults to `*`

**Returns:** `Table` instance for method chaining

### where()

Add WHERE conditions to filter results.

```typescript
// Simple equality
const activeUsers = await db.table("users").where("active", true).get();

// With operator
const adults = await db.table("users").where("age", ">=", 18).get();

// Multiple conditions
const results = await db
  .table("users")
  .where("active", true)
  .where("age", ">=", 18)
  .get();
```

**Parameters:**

- `column` (string): Column name
- `operator` (string, optional): Comparison operator (`=`, `!=`, `>`, `>=`, `<`, `<=`, `LIKE`, `IN`, etc.)
- `value` (any): Value to compare against

**Returns:** `Table` instance for method chaining

### whereIn()

Filter rows where column value is in a list of values.

```typescript
const users = await db.table("users").whereIn("id", [1, 2, 3, 4, 5]).get();
const posts = await db
  .table("posts")
  .whereIn("status", ["published", "featured"])
  .get();
```

**Parameters:**

- `column` (string): Column name
- `values` (any[]): Array of values to match

**Returns:** `Table` instance for method chaining

### whereNotIn()

Filter rows where column value is NOT in a list of values.

```typescript
const users = await db
  .table("users")
  .whereNotIn("status", ["banned", "deleted"])
  .get();
```

**Parameters:**

- `column` (string): Column name
- `values` (any[]): Array of values to exclude

**Returns:** `Table` instance for method chaining

### whereBetween()

Filter rows where column value is between two values.

```typescript
const adults = await db.table("users").whereBetween("age", [18, 65]).get();
const recentPosts = await db
  .table("posts")
  .whereBetween("created_at", ["2023-01-01", "2023-12-31"])
  .get();
```

**Parameters:**

- `column` (string): Column name
- `range` ([any, any]): Array with min and max values

**Returns:** `Table` instance for method chaining

### whereNull()

Filter rows where column value is NULL.

```typescript
const usersWithoutEmail = await db.table("users").whereNull("email").get();
```

**Parameters:**

- `column` (string): Column name

**Returns:** `Table` instance for method chaining

### whereNotNull()

Filter rows where column value is NOT NULL.

```typescript
const usersWithEmail = await db.table("users").whereNotNull("email").get();
```

**Parameters:**

- `column` (string): Column name

**Returns:** `Table` instance for method chaining

### orWhere()

Add OR WHERE conditions.

```typescript
const users = await db
  .table("users")
  .where("active", true)
  .orWhere("vip", true)
  .get();
```

**Parameters:**

- `column` (string): Column name
- `operator` (string, optional): Comparison operator
- `value` (any): Value to compare against

**Returns:** `Table` instance for method chaining

## Ordering and Limiting

### orderBy()

Sort results by one or more columns.

```typescript
// Single column ascending
const users = await db.table("users").orderBy("name").get();

// Single column descending
const users = await db.table("users").orderBy("created_at", "desc").get();

// Multiple columns
const users = await db
  .table("users")
  .orderBy("status", "asc")
  .orderBy("created_at", "desc")
  .get();
```

**Parameters:**

- `column` (string): Column name to sort by
- `direction` (string, optional): Sort direction (`asc` or `desc`). Defaults to `asc`

**Returns:** `Table` instance for method chaining

### limit()

Limit the number of results returned.

```typescript
const latestUsers = await db
  .table("users")
  .orderBy("created_at", "desc")
  .limit(10)
  .get();
```

**Parameters:**

- `count` (number): Maximum number of results to return

**Returns:** `Table` instance for method chaining

### offset()

Skip a number of results (for pagination).

```typescript
// Get results 11-20 (page 2, 10 per page)
const users = await db.table("users").orderBy("id").limit(10).offset(10).get();
```

**Parameters:**

- `count` (number): Number of results to skip

**Returns:** `Table` instance for method chaining

## Joins

### join()

Perform an inner join with another table.

```typescript
const usersWithPosts = await db
  .table("users")
  .join("posts", "users.id", "=", "posts.user_id")
  .select(["users.name", "posts.title"])
  .get();
```

**Parameters:**

- `table` (string): Table to join with
- `column1` (string): First column for join condition
- `operator` (string): Join operator (usually `=`)
- `column2` (string): Second column for join condition

**Returns:** `Table` instance for method chaining

### leftJoin()

Perform a left outer join.

```typescript
const usersWithOptionalPosts = await db
  .table("users")
  .leftJoin("posts", "users.id", "=", "posts.user_id")
  .select(["users.name", "posts.title"])
  .get();
```

### rightJoin()

Perform a right outer join.

```typescript
const postsWithUsers = await db
  .table("posts")
  .rightJoin("users", "posts.user_id", "=", "users.id")
  .select(["users.name", "posts.title"])
  .get();
```

## Data Retrieval

### get()

Execute the query and return all matching results.

```typescript
const users = await db.table("users").where("active", true).get();
// Returns: DatabaseRecord[]
```

**Returns:** `Promise<DatabaseRecord[]>` - Array of result objects

### first()

Execute the query and return only the first result.

```typescript
const user = await db.table("users").where("email", "john@example.com").first();
// Returns: DatabaseRecord | null
```

**Returns:** `Promise<DatabaseRecord | null>` - Single result object or null

### find()

Find a record by its primary key.

```typescript
const user = await db.table("users").find(1);
// Returns: DatabaseRecord | null
```

**Parameters:**

- `id` (any): Primary key value

**Returns:** `Promise<DatabaseRecord | null>` - Single result object or null

### count()

Get the count of matching records.

```typescript
const userCount = await db.table("users").where("active", true).count();
// Returns: number
```

**Returns:** `Promise<number>` - Count of matching records

## Data Modification

### insert()

Insert a new record into the table.

```typescript
// Insert single record
const result = await db.table("users").insert({
  name: "John Doe",
  email: "john@example.com",
  active: true,
});

// Insert multiple records
const results = await db.table("users").insert([
  { name: "Alice", email: "alice@example.com" },
  { name: "Bob", email: "bob@example.com" },
]);
```

**Parameters:**

- `data` (object | object[]): Single record object or array of record objects

**Returns:** `Promise<DatabaseRecord | DatabaseRecord[]>` - Inserted record(s) with generated IDs

### update()

Update existing records that match the current query conditions.

```typescript
const result = await db.table("users").where("id", 1).update({
  name: "Jane Doe",
  updated_at: new Date(),
});
```

**Parameters:**

- `data` (object): Object with column-value pairs to update

**Returns:** `Promise<number>` - Number of affected rows

### delete()

Delete records that match the current query conditions.

```typescript
// Delete specific user
const result = await db.table("users").where("id", 1).delete();

// Delete multiple users
const result = await db.table("users").where("active", false).delete();
```

**Returns:** `Promise<number>` - Number of deleted rows

## Raw Queries

### raw()

Execute a raw SQL query when you need more control.

```typescript
// Raw select query
const users = await db.raw("SELECT * FROM users WHERE age > ?", [18]);

// Raw insert query
const result = await db.raw("INSERT INTO users (name, email) VALUES (?, ?)", [
  "John",
  "john@example.com",
]);

// Raw query without parameters
const result = await db.raw("SHOW TABLES");
```

**Parameters:**

- `sql` (string): Raw SQL query string
- `bindings` (any[], optional): Array of parameter values for prepared statements

**Returns:** `Promise<any>` - Raw database results

## Transactions

### transaction()

Execute multiple queries within a database transaction.

```typescript
await db.transaction(async (trx) => {
  // Create user
  const user = await trx.table("users").insert({
    name: "John Doe",
    email: "john@example.com",
  });

  // Create user profile
  await trx.table("user_profiles").insert({
    user_id: user.id,
    bio: "Software developer",
  });

  // Both queries are committed together
  // If any query fails, all are rolled back
});
```

**Parameters:**

- `callback` (function): Async function that receives a transaction instance

**Returns:** `Promise<any>` - Result of the callback function

## Advanced Features

### chunk()

Process large datasets in chunks to manage memory usage.

```typescript
await db.table("users").chunk(100, async (users) => {
  // Process each chunk of 100 users
  for (const user of users) {
    await processUser(user);
  }
});
```

**Parameters:**

- `size` (number): Number of records per chunk
- `callback` (function): Function to process each chunk

**Returns:** `Promise<void>`

### exists()

Check if any records match the current query conditions.

```typescript
const hasActiveUsers = await db.table("users").where("active", true).exists();
// Returns: boolean
```

**Returns:** `Promise<boolean>` - True if matching records exist

## Examples

### Complete CRUD Operations

```typescript
import { Dbcube } from "dbcube";

const dbcube = new Dbcube();
await dbcube.init();
const db = dbcube.database("myapp");

// Create
const user = await db.table("users").insert({
  name: "John Doe",
  email: "john@example.com",
  active: true,
});

// Read
const users = await db
  .table("users")
  .select(["id", "name", "email"])
  .where("active", true)
  .orderBy("name")
  .get();

const user = await db.table("users").find(1);

// Update
await db.table("users").where("id", 1).update({ name: "Jane Doe" });

// Delete
await db.table("users").where("id", 1).delete();
```

### Complex Query Example

```typescript
const results = await db
  .table("users")
  .select(["users.name", "posts.title", "categories.name as category"])
  .join("posts", "users.id", "=", "posts.user_id")
  .join("categories", "posts.category_id", "=", "categories.id")
  .where("users.active", true)
  .where("posts.published", true)
  .whereIn("categories.slug", ["tech", "programming", "typescript"])
  .orderBy("posts.created_at", "desc")
  .limit(20)
  .get();
```

### Pagination Example

```typescript
async function getPaginatedUsers(page = 1, perPage = 10) {
  const offset = (page - 1) * perPage;

  const users = await db
    .table("users")
    .select(["id", "name", "email", "created_at"])
    .where("active", true)
    .orderBy("created_at", "desc")
    .limit(perPage)
    .offset(offset)
    .get();

  const total = await db.table("users").where("active", true).count();

  return {
    data: users,
    pagination: {
      page,
      perPage,
      total,
      totalPages: Math.ceil(total / perPage),
    },
  };
}
```
