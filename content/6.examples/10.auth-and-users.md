The user/account layer almost every app needs: register, log in, manage sessions
and profiles — typed end to end. (Password hashing uses your library of choice;
Dbcube just stores and queries.)

## The types

```ts
// dbcube/types.ts  (generated)
export interface User {
  id: number;
  email: string;
  password_hash: string;
  name: string;
  role: "user" | "admin";
  verified: boolean;
  deleted_at: string | null;
  created_at: string;
}

export interface Session {
  id: number;
  user_id: number;
  token: string;
  expires_at: string;
}
```

## Registration

`email` is `UNIQUE` in the schema, so a duplicate insert fails — we check first
for a friendly error.

```ts
import { dbcube } from "dbcube";
import bcrypt from "bcryptjs";
import { randomUUID } from "node:crypto";
import type { User } from "./dbcube/types";

const db = dbcube.database("app");

export async function register(email: string, password: string, name: string): Promise<User> {
  const existing = await db.table<User>("users").where("email", "=", email).first();
  if (existing) throw new Error("Email already registered");

  const [user] = await db.table<User>("users").insert([
    {
      email,
      password_hash: await bcrypt.hash(password, 12),
      name,
      role: "user",
      verified: false,
    },
  ]);
  return user;
}
```

## Login → issue a session

```ts
export async function login(email: string, password: string): Promise<string> {
  const user = await db.table<User>("users")
    .where("email", "=", email)
    .whereNull("deleted_at")        // soft-deleted users can't log in
    .first();

  if (!user || !(await bcrypt.compare(password, user.password_hash))) {
    throw new Error("Invalid credentials");
  }

  const token = randomUUID();
  const expires = new Date(Date.now() + 7 * 24 * 3600 * 1000).toISOString();
  await db.table<Session>("sessions").insert([
    { user_id: user.id, token, expires_at: expires },
  ]);
  return token;
}
```

## Authenticating a request (session lookup)

```ts
export async function userFromToken(token: string): Promise<User | null> {
  const session = await db.table<Session>("sessions")
    .where("token", "=", token)
    .where("expires_at", ">", new Date().toISOString())
    .first();
  if (!session) return null;

  return db.table<User>("users")
    .where("id", "=", session.user_id)
    .whereNull("deleted_at")
    .first();
}
```

## Logout & cleaning expired sessions

```ts
export async function logout(token: string): Promise<void> {
  await db.table<Session>("sessions").where("token", "=", token).delete();
}

// Run periodically (cron / scheduled job)
export async function purgeExpiredSessions(): Promise<void> {
  await db.table<Session>("sessions")
    .where("expires_at", "<", new Date().toISOString())
    .delete();
}
```

## Profile update & email verification

```ts
export async function updateProfile(userId: number, patch: Partial<Pick<User, "name" | "email">>) {
  await db.table<User>("users").where("id", "=", userId).update(patch);
}

export async function markVerified(userId: number) {
  await db.table<User>("users").where("id", "=", userId).update({ verified: true });
}
```

## Soft delete (keep the row, hide the user)

```ts
export async function softDelete(userId: number): Promise<void> {
  await db.transaction(async (trx) => {
    await trx.table<User>("users")
      .where("id", "=", userId)
      .update({ deleted_at: new Date().toISOString() });
    // kill all active sessions in the same atomic step
    await trx.table<Session>("sessions").where("user_id", "=", userId).delete();
  });
}
```

## Admin: list users with a role filter and pagination

```ts
export async function listUsers(page = 1, role?: User["role"]) {
  let q = db.table<User>("users").whereNull("deleted_at");
  if (role) q = q.where("role", "=", role);
  return q.orderBy("created_at", "DESC").paginate(page, 25);
}
```

::callout{type="info"}
Dbcube stores and queries; it doesn't hash passwords or sign tokens. Use a
dedicated library (`bcryptjs`/`argon2` for hashing, `jose`/`jsonwebtoken` for
JWTs) and keep secrets in environment variables.
::

::callout{type="warning"}
Make `email` and `sessions.token` **UNIQUE** in your `.cube` schema. That pushes
the uniqueness guarantee into the database, so two concurrent registrations with
the same email can't both succeed.
::
