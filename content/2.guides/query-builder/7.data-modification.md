## Data Insertion

### insert() - Insert Records

```typescript
// Insert a single record
const user = await db.table("users").insert({
  name: "John Doe",
  email: "john@example.com",
  age: 28,
  active: true,
  registration_date: new Date(),
});

console.log(`User created with ID: ${user.id}`);

// Insert multiple records
const users = await db.table("users").insert([
  {
    name: "Anna Garcia",
    email: "anna@example.com",
    age: 25,
  },
  {
    name: "Carlos Lopez",
    email: "carlos@example.com",
    age: 32,
  },
  {
    name: "Maria Rodriguez",
    email: "maria@example.com",
    age: 29,
  },
]);

console.log(`${users.length} users created`);
```

### insertGetId() - Insert and Get ID

```typescript
// Insert and get only the generated ID
const newId = await db.table("posts").insertGetId({
  title: "My New Post",
  content: "Post content...",
  user_id: 1,
  status: "draft",
});

console.log(`Post created with ID: ${newId}`);
```

### insertIgnore() - Insert Ignoring Duplicates

```typescript
// Insert ignoring duplicate errors
const result = await db.table("tags").insertIgnore([
  { name: "typescript" },
  { name: "typescript" },
  { name: "typescript" }, // Will be ignored if it already exists
]);

console.log(`${result.length} tags inserted`);
```

## Data Updates

### update() - Update Records

```typescript
// Update a record by ID
const affectedRows = await db.table("users").where("id", 1).update({
  name: "John Carlos Doe",
  updated_at: new Date(),
});

console.log(`${affectedRows} user(s) updated`);

// Update multiple records
const result = await db
  .table("posts")
  .where("status", "draft")
  .where("user_id", 1)
  .update({
    status: "published",
    published_at: new Date(),
  });

console.log(`${result} posts published`);

// Update with SQL expressions
const increment = await db
  .table("products")
  .where("category_id", 1)
  .update({
    price: db.raw("price * 1.1"), // Increase by 10%
    stock: db.raw("stock - 1"),
    updated_at: db.raw("NOW()"),
  });
```

### updateOrInsert() - Update or Insert

```typescript
// Search by email, update if exists, insert if not
const user = await db.table("users").updateOrInsert(
  { email: "new@example.com" }, // Search condition
  {
    name: "New User",
    age: 30,
    active: true,
    updated_at: new Date(),
  }
);
```

### increment() and decrement() - Increment/Decrement

```typescript
// Increment visit counter
await db.table("posts").where("id", 1).increment("views", 1);

// Increment multiple columns
await db.table("users").where("id", 1).increment({
  points: 10,
  level: 1,
});

// Decrement stock
await db.table("products").where("id", 1).decrement("stock", 5);

// Decrement with minimum value
await db.table("users").where("id", 1).decrement("credits", 100, 0); // Don't go below 0
```

## Data Deletion

### delete() - Delete Records

```typescript
// Delete a specific record
const deleted = await db.table("users").where("id", 1).delete();

console.log(`${deleted} user(s) deleted`);

// Delete multiple records
const result = await db
  .table("posts")
  .where("status", "draft")
  .where("created_at", "<", "2022-01-01")
  .delete();

// Delete with complex conditions
const spam = await db
  .table("comments")
  .where("approved", false)
  .where(function () {
    this.whereLike("content", "%spam%")
      .orWhereLike("content", "%viagra%")
      .orWhere("author_email", "LIKE", "%@tempmail.%");
  })
  .delete();
```

### truncate() - Empty Table

```typescript
// Empty table completely (faster than DELETE)
await db.table("temp_logs").truncate();

// Reset auto-increment
await db.table("user_sessions").truncate();
```

## UPSERT Operations

### upsert() - Insert or Update

```typescript
// UPSERT a single record
const result = await db.table("settings").upsert(
  {
    key: "site_theme",
    value: "dark",
    updated_at: new Date(),
  },
  ["key"] // Conflict column(s)
);

// Multiple UPSERT
const stats = await db.table("daily_stats").upsert(
  [
    {
      date: "2023-12-01",
      visits: 1250,
      sales: 85,
      revenue: 2400.5,
    },
    {
      date: "2023-12-02",
      visits: 1340,
      sales: 92,
      revenue: 2650.75,
    },
  ],
  ["date"], // Conflict columns
  ["visits", "sales", "revenue"] // Columns to update
);
```

### onDuplicateKeyUpdate() - MySQL

```typescript
// MySQL: ON DUPLICATE KEY UPDATE
const result = await db
  .table("counters")
  .insert({
    name: "page_views",
    value: 1,
  })
  .onDuplicateKeyUpdate({
    value: db.raw("value + 1"),
    updated_at: db.raw("NOW()"),
  });
```

## Transactions

### transaction() - Atomic Operations

```typescript
// Simple transaction
await db.transaction(async (trx) => {
  // Create user
  const user = await trx.table("users").insert({
    name: "John Doe",
    email: "john@example.com",
  });

  // Create profile
  await trx.table("user_profiles").insert({
    user_id: user.id,
    bio: "Software developer",
    location: "Madrid, Spain",
  });

  // Assign default role
  await trx.table("user_roles").insert({
    user_id: user.id,
    role_id: 2, // Normal user
  });

  console.log("User created successfully");
});

// Transaction with error handling
try {
  await db.transaction(async (trx) => {
    // Money transfer operation
    const sourceAccount = await trx
      .table("accounts")
      .where("id", sourceAccountId)
      .first();

    if (sourceAccount.balance < amount) {
      throw new Error("Insufficient balance");
    }

    // Debit source account
    await trx
      .table("accounts")
      .where("id", sourceAccountId)
      .decrement("balance", amount);

    // Credit destination account
    await trx
      .table("accounts")
      .where("id", destinationAccountId)
      .increment("balance", amount);

    // Record transaction
    await trx.table("transactions").insert({
      source_account: sourceAccountId,
      destination_account: destinationAccountId,
      amount,
      type: "transfer",
      date: new Date(),
    });
  });

  console.log("Transfer completed");
} catch (error) {
  console.error("Transfer error:", error.message);
}
```

### savepoint() - Save Points

```typescript
// Use savepoints for granular control
await db.transaction(async (trx) => {
  // Main operation
  const order = await trx.table("orders").insert({
    user_id: 1,
    total: 150.0,
    status: "pending",
  });

  // Savepoint before optional operations
  await trx.savepoint("before_discounts");

  try {
    // Try to apply discounts
    await trx.table("applied_discounts").insert({
      order_id: order.id,
      code: "DISCOUNT10",
      amount: 15.0,
    });

    await trx.table("orders").where("id", order.id).update({ total: 135.0 });
  } catch (error) {
    // Rollback only to savepoint
    await trx.rollbackTo("before_discounts");
    console.log("Could not apply discount, continuing without it");
  }

  // Order is created anyway
  console.log("Order created:", order.id);
});
```

## Practical Examples

### 1. E-commerce System

```typescript
// Process complete order
async function processOrder(userData: any, items: any[]) {
  return await db.transaction(async (trx) => {
    // 1. Create or update user
    const user = await trx.table("users").updateOrInsert(
      { email: userData.email },
      {
        name: userData.name,
        phone: userData.phone,
        address: userData.address,
        updated_at: new Date(),
      }
    );

    // 2. Calculate total and verify stock
    let orderTotal = 0;
    for (const item of items) {
      const product = await trx
        .table("products")
        .where("id", item.product_id)
        .first();

      if (product.stock < item.quantity) {
        throw new Error(`Insufficient stock for ${product.name}`);
      }

      orderTotal += product.price * item.quantity;
    }

    // 3. Create order
    const order = await trx.table("orders").insert({
      user_id: user.id,
      total: orderTotal,
      status: "pending",
      order_date: new Date(),
    });

    // 4. Create order items and update stock
    for (const item of items) {
      await trx.table("order_items").insert({
        order_id: order.id,
        product_id: item.product_id,
        quantity: item.quantity,
        unit_price: item.price,
      });

      await trx
        .table("products")
        .where("id", item.product_id)
        .decrement("stock", item.quantity);
    }

    return order;
  });
}
```

### 2. Points and Achievements System

```typescript
// Award points and check achievements
async function awardPoints(userId: number, points: number, reason: string) {
  return await db.transaction(async (trx) => {
    // Update user points
    await trx
      .table("users")
      .where("id", userId)
      .increment("total_points", points);

    // Record points transaction
    await trx.table("points_history").insert({
      user_id: userId,
      points,
      reason,
      date: new Date(),
    });

    // Get current points
    const user = await trx
      .table("users")
      .select("total_points")
      .where("id", userId)
      .first();

    // Check achievements
    const availableAchievements = await trx
      .table("achievements")
      .where("required_points", "<=", user.total_points)
      .whereNotIn("id", function () {
        this.select("achievement_id")
          .from("user_achievements")
          .where("user_id", userId);
      })
      .get();

    // Award new achievements
    const newAchievements = [];
    for (const achievement of availableAchievements) {
      await trx.table("user_achievements").insert({
        user_id: userId,
        achievement_id: achievement.id,
        achieved_at: new Date(),
      });
      newAchievements.push(achievement);
    }

    return {
      pointsAwarded: points,
      currentPoints: user.total_points,
      newAchievements,
    };
  });
}
```

### 3. Audit System

```typescript
// Wrapper for operations with audit
class AuditableTable {
  constructor(
    private db: any,
    private tableName: string
  ) {}

  async insert(data: any, userId: number) {
    return await this.db.transaction(async (trx: any) => {
      const record = await trx.table(this.tableName).insert(data);

      await trx.table("audit_log").insert({
        table_name: this.tableName,
        record_id: record.id,
        action: "INSERT",
        new_data: JSON.stringify(data),
        user_id: userId,
        timestamp: new Date(),
        ip_address: this.getClientIP(),
      });

      return record;
    });
  }

  async update(id: number, data: any, userId: number) {
    return await this.db.transaction(async (trx: any) => {
      const oldData = await trx.table(this.tableName).where("id", id).first();

      const result = await trx
        .table(this.tableName)
        .where("id", id)
        .update(data);

      if (result > 0) {
        await trx.table("audit_log").insert({
          table_name: this.tableName,
          record_id: id,
          action: "UPDATE",
          old_data: JSON.stringify(oldData),
          new_data: JSON.stringify(data),
          user_id: userId,
          timestamp: new Date(),
          ip_address: this.getClientIP(),
        });
      }

      return result;
    });
  }

  private getClientIP() {
    // Logic to get client IP
    return "192.168.1.1";
  }
}

// Usage
const auditableUsers = new AuditableTable(db, "users");

// Operations are automatically logged
const newUser = await auditableUsers.insert(
  {
    name: "John Doe",
    email: "john@example.com",
  },
  1
); // ID of user performing the action
```

## Best Practices

### 1. Data Validation

```typescript
// Always validate before inserting
function validateUser(data: any) {
  const errors = [];

  if (!data.name || data.name.length < 2) {
    errors.push("Name must be at least 2 characters");
  }

  if (!data.email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(data.email)) {
    errors.push("Invalid email");
  }

  if (data.age && (data.age < 0 || data.age > 120)) {
    errors.push("Invalid age");
  }

  return errors;
}

async function createUser(data: any) {
  const errors = validateUser(data);
  if (errors.length > 0) {
    throw new Error(`Invalid data: ${errors.join(", ")}`);
  }

  return await db.table("users").insert(data);
}
```

### 2. Concurrency Handling

```typescript
// Optimistic concurrency control with timestamps
async function updateProductSafely(
  id: number,
  data: any,
  expectedTimestamp: string
) {
  return await db.transaction(async (trx) => {
    const product = await trx.table("products").where("id", id).first();

    if (product.updated_at !== expectedTimestamp) {
      throw new Error("Product has been modified by another user");
    }

    return await trx
      .table("products")
      .where("id", id)
      .update({
        ...data,
        updated_at: new Date(),
      });
  });
}
```

### 3. Batch Operations

```typescript
// Efficient batch insertion
async function insertInBatches(data: any[], batchSize: number = 100) {
  const results = [];

  for (let i = 0; i < data.length; i += batchSize) {
    const batch = data.slice(i, i + batchSize);
    const result = await db.table("table").insert(batch);
    results.push(...result);

    console.log(`Processed batch ${Math.floor(i / batchSize) + 1}`);
  }

  return results;
}
```
