## Configuration File

Dbcube looks for a `dbcube.config.js` file in your project root. This file exports a function that receives a configuration object.

```typescript
// dbcube.config.js
module.exports = (config) => {
  config.addDatabase({
    name: "myapp",
    type: "mysql",
    host: "localhost",
    port: 3306,
    user: "root",
    password: "password",
    database: "myapp_db"
  });
};
```

## Configuration Parameters

Each database connection requires these parameters:

| Parameter | Type | Required | Description |
|-----------|------|----------|-------------|
| `name` | string | Yes | Unique identifier for this database connection |
| `type` | string | Yes | Database type: `mysql`, `postgres`, `sqlite`, or `mongodb` |
| `host` | string | Yes* | Database server hostname |
| `port` | number | Yes* | Database server port |
| `user` | string | Yes* | Database username |
| `password` | string | Yes* | Database password |
| `database` | string | Yes | Database name |

\* Not required for SQLite (file-based database)

## Database Types

### MySQL

```typescript
config.addDatabase({
  name: "mysql_db",
  type: "mysql",
  host: "localhost",
  port: 3306,
  database: "myapp",
  user: "root",
  password: "password"
});
```

Default port: `3306`

### PostgreSQL

```typescript
config.addDatabase({
  name: "postgres_db",
  type: "postgres",
  host: "localhost",
  port: 5432,
  database: "myapp",
  user: "postgres",
  password: "password"
});
```

Default port: `5432`

### SQLite

```typescript
config.addDatabase({
  name: "sqlite_db",
  type: "sqlite",
  database: "myapp"  // Creates .dbcube/myapp.db
});
```

SQLite databases are stored in the `.dbcube/` directory in your project root.

### MongoDB

```typescript
config.addDatabase({
  name: "mongo_db",
  type: "mongodb",
  host: "mongodb://localhost:27017",
  database: "myapp",
  user: "admin",
  password: "password"
});
```

Default port: `27017`

## Multiple Databases

You can configure multiple database connections in the same file:

```typescript
// dbcube.config.js
module.exports = (config) => {
  // Primary database
  config.addDatabase({
    name: "primary",
    type: "mysql",
    host: "localhost",
    port: 3306,
    database: "app_primary",
    user: "root",
    password: "password"
  });

  // Analytics database
  config.addDatabase({
    name: "analytics",
    type: "postgres",
    host: "localhost",
    port: 5432,
    database: "analytics_db",
    user: "postgres",
    password: "password"
  });

  // Cache database
  config.addDatabase({
    name: "cache",
    type: "sqlite",
    database: "cache"
  });
};
```

Then use them in your code:

```typescript
import Database from '@dbcube/query-builder';

const primaryDb = new Database("primary");
const users = await primaryDb.table('users').get();

const analyticsDb = new Database("analytics");
const stats = await analyticsDb.table('stats').get();
```

## Environment Variables

Use environment variables to keep sensitive data secure. Create a `.env` file:

```bash
# .env
DB_HOST=localhost
DB_PORT=3306
DB_NAME=myapp_db
DB_USER=root
DB_PASSWORD=password
```

Then use in your configuration:

```typescript
// dbcube.config.js
require('dotenv').config();

module.exports = (config) => {
  config.addDatabase({
    name: "myapp",
    type: "mysql",
    host: process.env.DB_HOST,
    port: parseInt(process.env.DB_PORT),
    database: process.env.DB_NAME,
    user: process.env.DB_USER,
    password: process.env.DB_PASSWORD
  });
};
```

### Environment-Specific Configuration

Create different configurations for different environments:

```typescript
// dbcube.config.js
require('dotenv').config();

module.exports = (config) => {
  const env = process.env.NODE_ENV || 'development';

  if (env === 'development') {
    config.addDatabase({
      name: "main",
      type: "mysql",
      host: "localhost",
      port: 3306,
      database: "myapp_dev",
      user: "root",
      password: "password"
    });
  } else if (env === 'production') {
    config.addDatabase({
      name: "main",
      type: "mysql",
      host: process.env.DB_HOST,
      port: parseInt(process.env.DB_PORT),
      database: process.env.DB_NAME,
      user: process.env.DB_USER,
      password: process.env.DB_PASSWORD
    });
  } else if (env === 'test') {
    config.addDatabase({
      name: "main",
      type: "sqlite",
      database: "test"
    });
  }
};
```

Create separate `.env` files for each environment:

```bash
# .env.development
DB_HOST=localhost
DB_PORT=3306
DB_NAME=myapp_dev
DB_USER=root
DB_PASSWORD=password

# .env.production
DB_HOST=prod-db.example.com
DB_PORT=3306
DB_NAME=myapp_prod
DB_USER=app_user
DB_PASSWORD=secure_production_password

# .env.test
DB_HOST=localhost
DB_PORT=3306
DB_NAME=myapp_test
DB_USER=root
DB_PASSWORD=password
```

Load the appropriate file:

```typescript
const env = process.env.NODE_ENV || 'development';
require('dotenv').config({ path: `.env.${env}` });
```

## Validation and Errors

Dbcube validates your configuration and provides clear error messages.

**Missing required fields:**

```typescript
// ❌ This will error
config.addDatabase({
  name: "invalid",
  type: "mysql",
  host: "localhost"
  // Missing: port, user, password, database
});
```

Error: `Missing required configuration fields: port, user, password, database`

**Invalid database type:**

```typescript
// ❌ This will error
config.addDatabase({
  name: "invalid",
  type: "mongodb-atlas",  // Not supported
  host: "localhost",
  port: 3306,
  database: "myapp",
  user: "root",
  password: "password"
});
```

Error: `Invalid database type 'mongodb-atlas'. Supported types: mysql, postgres, sqlite, mongodb`

**Duplicate database names:**

```typescript
// ❌ This will error
config.addDatabase({ name: "main", type: "mysql", ... });
config.addDatabase({ name: "main", type: "postgres", ... });
```

Error: `Database with name 'main' already exists`

## Troubleshooting

**Connection refused** (`ECONNREFUSED`)

Verify database server is running and check host/port settings:

```bash
# Check if MySQL is running
sudo systemctl status mysql

# Check if PostgreSQL is running
sudo systemctl status postgresql

# Test connection manually
mysql -h localhost -u root -p
```

**Authentication failed** (`Access denied`)

Verify username and password are correct:

```bash
# Check MySQL user
mysql -u root -p -e "SELECT user, host FROM mysql.user;"

# Grant permissions
mysql -u root -p -e "GRANT ALL PRIVILEGES ON myapp.* TO 'root'@'localhost';"
```

**Database not found** (`Unknown database`)

Create the database first:

```bash
# Create database with CLI
dbcube run database:create

# Or create manually
mysql -u root -p -e "CREATE DATABASE myapp;"
```

**SQLite directory not found**

The `.dbcube` directory is created automatically. Ensure you have write permissions:

```bash
mkdir .dbcube
```

**MongoDB connection string error**

Verify the connection string format includes the protocol:

```typescript
// ✓ Correct
host: "mongodb://localhost:27017"

// ✗ Incorrect
host: "localhost:27017"  // Missing protocol
```

## Security Best Practices

**Never commit credentials** - Add to `.gitignore`:

```bash
.env
.env.*
.dbcube/
dbcube.config.js  # If it contains credentials
```

**Use environment variables:**

```typescript
// ✓ Good
password: process.env.DB_PASSWORD

// ✗ Bad
password: "my_password_123"
```

**Use different credentials per environment** - Separate credentials for development, staging, and production.

**Rotate passwords regularly** - Change database passwords periodically, especially after team member changes.
